foederatus_special_contract = {
	display_mode = radiobutton
	is_shown = {
		#Should only show up for active foederatus; installing a foederatus is an interaction
		#and you shouldn't be able to convert normal vassals to foederatus
		scope:vassal = { vassal_contract_has_flag = foederatus_installed }
	}
	obligation_levels = {
		foederatus_none = {
			default = yes
			position = { 0 0 }
			
			ai_liege_desire = {}
			ai_vassal_desire = {}
		}
		foederatus_installed = {
			is_valid = { scope:vassal = { valid_government_to_have_foederatus_trigger = yes } }
			parent = foederatus_none
			position = { 1 0 }
			
			min_levies = 0.50
			tax_factor = -1.0
			
			vassal_modifier = {
				levy_size = 0.25
				levy_reinforcement_rate = 0.1
				development_growth_factor = 0.10
			}
			
			flag = religiously_protected
			flag = vassal_contract_cannot_revoke_titles
			flag = locked_to_partition
			flag = cannot_retract_vassals
			
			flag = foederatus_installed
			
			ai_liege_desire = {}
			ai_vassal_desire = {}
		}
	}
}
