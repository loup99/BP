imperial_autonomy = {
	display_mode = checkbox
	is_shown = {
		scope:vassal = {
			top_liege = { has_government = bureaucratic_government }
			scope:vassal.primary_title.tier >= tier_county
		}
	}
	obligation_levels = {
		imperial_autonomy_none = {
			default = yes
			position = { 0 0 }
			
			ai_liege_desire = @ai_standard_liege_desire
			ai_vassal_desire = 0
		}
		imperial_autonomy_full = {
			is_valid = {
				scope:vassal = {
					custom_description = {
						text = imperial_autonomy_full_CONDITION
						OR = {
							vassal_contract_has_flag = imperial_autonomy_full
							has_character_flag = imperial_autonomy_override
							prestige_level >= 4
						}
					}	
				}
			}
			parent = imperial_autonomy_none
			position = { 1 0 }
			
			vassal_opinion = 10
			
			liege_modifier = {
				monthly_prestige_gain_mult = -0.05
				powerful_vassal_opinion = -1
			}
			
			vassal_modifier = { fellow_vassal_opinion = -10 }
			
			flag = imperial_autonomy_full
			
			ai_liege_desire = 0
			ai_vassal_desire = 10
		}
	}
}
