#Scripted Modifiers used for buildings

#Throws errors if not including the is_landed check;
#TODO: Review after 1.5 to see if Paradox fixes this
ai_general_building_modifier = {
	modifier = {
		add = -2
		AND = {
			scope:holder = { is_landed = yes }
			NOT = { this = scope:holder.capital_province }
		}
	}
}
