#LT-Rascek: Duplicate of Tribal; brought in for easy integration if/when Paradox implements Nomadic Governance
nomad_government = {
	rulers_should_have_dynasty = yes
	affected_by_development = no
	regiments_prestige_as_gold = yes
	supply_limit_mult_for_others = -0.5
	primary_holding = tribal_holding
	required_county_holdings = { tribal_holding }
	
	primary_culture_groups = { mongolic_group turkic_group }
	
	create_cadet_branches = yes
	rulers_should_have_dynasty = yes
	
	required_county_holdings = { castle_holding city_holding church_holding }
	always_use_patronym = yes
	
	vassal_contract = {
		nomadic_government_obligations
		foederatus_special_contract
		tributary_special_contract
	}
	
	opinion_of_liege = {
		scope:vassal = {
			if = {
				limit = { NOT = { is_allied_to = scope:liege } }
				if = {
					limit = { is_powerful_vassal = yes }
					value = -30
				}
				else = { value = -15 }
			}
		}
	}
	opinion_of_liege_desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:vassal = {
						NOT = { is_allied_to = scope:liege }
						is_powerful_vassal = yes
					}
				}
				desc = "GOVERNMENT_CLAN_NOT_ALLIED_POWERFUL"
			}
			triggered_desc = {
				trigger = {
					scope:vassal = {
						NOT = { is_allied_to = scope:liege }
						is_powerful_vassal = no
					}
				}
				desc = "GOVERNMENT_CLAN_NOT_ALLIED"
			}
		}
	}
	
	character_modifier = {
		title_creation_cost_mult = -0.5
		army_maintenance_mult = -0.5
		development_growth_factor = -0.75
		
		feudal_government_vassal_opinion = -20
		clan_government_vassal_opinion = -20
		republic_government_vassal_opinion = -20
		
		men_at_arms_cap = 1
		levy_size = 0.5
		domain_tax_mult = -0.5
		
		ai_war_chance = 0.50
	}
	
	flag = government_can_raid_rule
	flag = cannot_advance_beyond_tribal_era
	flag = nomad_government_character_modifier
	
	color ={ 201 179 36 }
}
