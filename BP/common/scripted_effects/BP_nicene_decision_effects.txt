issue_henotikon_effects = {
	faith:nicene.religion = { #All Christian faiths
		#All branches of Christianity that are Eutychian, Monophysite, or Nestorian are condemned
		#And no longer in union with Nicea, if they were otherwise
		every_faith = {
			limit = {
				this.religion = faith:nicene.religion
				OR = {
					has_doctrine = doctrine_christian_christology_eutychianism
					has_doctrine = doctrine_christian_christology_nestorianism
					has_doctrine = doctrine_christian_christology_monophysitism
				}
				NOT = { this = root.faith }
			}
			if = {
				limit = { has_doctrine = special_doctrine_ecumenical_christian }
				remove_doctrine = special_doctrine_ecumenical_christian
			}
			change_fervor = -20
		}
		#Nicene fervor drops sharply initially
		every_faith = {
			limit = { this = root.faith }
			change_fervor = -10
		}
		#All counties under the emperor get an opinion bonus
		#TODO: Restrict to counties only having religions without specified doctrines
		every_ruler = {
			limit = { has_title = title:e_byzantium }
			every_sub_realm_county = {
				add_county_modifier = { modifier = henotikon_county_opinion }
			}
		}
		#Gets an opinion bonus for mia, dyo, and monothelites in the Empire
		every_ruler = {
			limit = {
				OR = {
					has_faith = faith:nicene
					has_faith = faith:coptic
					has_faith = faith:armenian_apostolic
					has_faith = faith:georgian_orthodox
				}
				is_vassal_or_below_of = scope:henotikon_issuer
			}
			add_opinion = {
				modifier = henotikon_issued_vassal_opinion
				target = scope:henotikon_issuer
			}
		}
		#Nicene realm priests not under the emperor's thumb are pissed
		every_ruler = {
			limit = {
				has_faith = faith:nicene
				is_clergy = yes
				NOT = { is_vassal_or_below_of = scope:henotikon_issuer }
			}
			add_opinion = {
				modifier = henotikon_issued_clerical_opinion
				target = scope:henotikon_issuer
			}
		}
		#Independent secular rulers are less pissed
		every_ruler = {
			limit = {
				has_faith = faith:nicene
				is_clergy = no
				NOT = { is_vassal_or_below_of = scope:henotikon_issuer }
			}
			add_opinion = {
				modifier = henotikon_issued_secular_opinion
				target = scope:henotikon_issuer
			}
		}
	}
}#end:issue_henotikon_effects
