### Government Conversion Decisions
convert_whole_realm_to_high_tribalism_effect = {
	hidden_effect = { remove_all_realm_laws = yes }
	#Change government
	change_government = high_tribal_government
	#Setup Centralization
	if = {
		limit = { NOT = { has_realm_law = high_tribal_organization_0 } }
		add_realm_law_skip_effects = high_tribal_organization_0
	}
}

convert_to_bureaucratic_effect = {
	hidden_effect = { remove_all_realm_laws = yes }
	change_government = bureaucratic_government
	every_vassal_or_below = {
		limit = { has_government = subroman_government }
		change_government = gubernatorial_government
	}
	set_taxation_law_to_value = { VALUE = 2 }
	every_held_title = {
		limit = { this.tier = tier_empire }
		remove_all_title_laws = yes
		if = {
			limit = { NOT = { has_title_law = imperial_elective_succession_law } }
			add_title_law = imperial_elective_succession_law
		}
	}
}
