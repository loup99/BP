#Applies both bureaucratic and gubernatorial taxation based on grantor realm law
get_taxation_law = {
	if = {
		limit = { scope:previous_holder = { has_realm_law = bureaucratic_taxation_0 } }
		add_realm_law_skip_effects = bureaucratic_taxation_0
	}
	else_if = {
		limit = { scope:previous_holder = { has_realm_law = bureaucratic_taxation_1 } }
		add_realm_law_skip_effects = bureaucratic_taxation_1
	}
	else_if = {
		limit = { scope:previous_holder = { has_realm_law = bureaucratic_taxation_2 } }
		add_realm_law_skip_effects = bureaucratic_taxation_2
	}
	else_if = {
		limit = { scope:previous_holder = { has_realm_law = bureaucratic_taxation_3 } }
		add_realm_law_skip_effects = bureaucratic_taxation_3
	}
	else_if = {
		limit = { scope:previous_holder = { has_realm_law = bureaucratic_taxation_4 } }
		add_realm_law_skip_effects = bureaucratic_taxation_4
	}
}
apply_taxation_down_tree = {
	every_vassal_or_below = {
		if = {
			limit = {
				has_government = bureaucratic_government
				NOT = { has_realm_law = bureaucratic_taxation_$VALUE$ }
			}
			add_realm_law_skip_effects = bureaucratic_taxation_$VALUE$
		}
	}
}
#E.g., independency or invalid liege govt
revert_bureaucratic_government_if_necessary = { #Top-level liege scope
	if = {
		limit = { valid_bureaucratic_government_state_trigger = no }
		change_government = subroman_government
		add_realm_law_skip_effects = subroman_decay_4
		#Because we change the top dog's government, the rest should now be invalid
		revert_vassal_bureaucratic_government_if_necessary = yes
	}
}
revert_vassal_bureaucratic_government_if_necessary = { #Top level liege scope
	every_vassal_or_below = {
		limit = {
			has_government = bureaucratic_government
			valid_bureaucratic_government_state_trigger = no
		}
		change_government = subroman_government
		add_realm_law_skip_effects = subroman_decay_4
	}
}
#Decrease realm law (from liberty faction and liberty faction event scopes)
decrease_realm_law = {
	#Crown Authority Cases
	if = {
		limit = { has_realm_law = crown_authority_1 }
		add_realm_law = crown_authority_0
	}
	if = {
		limit = { has_realm_law = crown_authority_2 }
		add_realm_law = crown_authority_1
	}
	if = {
		limit = { has_realm_law = crown_authority_3 }
		add_realm_law = crown_authority_2
	}
	#Prefeudal Cases
	if = {
		limit = { has_realm_law = prefeudal_authority_1 }
		add_realm_law = prefeudal_authority_0
	}
	if = {
		limit = { has_realm_law = prefeudal_authority_2 }
		add_realm_law = prefeudal_authority_1
	}
	if = {
		limit = { has_realm_law = prefeudal_authority_3 }
		add_realm_law = prefeudal_authority_2
	}
	#High Tribal Organization Cases
	if = {
		limit = { has_realm_law = high_tribal_organization_1 }
		add_realm_law = high_tribal_organization_0
	}
	if = {
		limit = { has_realm_law = high_tribal_organization_2 }
		add_realm_law = high_tribal_organization_1
	}
	if = {
		limit = { has_realm_law = high_tribal_organization_3 }
		add_realm_law = high_tribal_organization_2
	}
	#For countries with subroman organization
	if = {
		limit = { has_realm_law = subroman_decay_1 }
		add_realm_law = subroman_decay_1
	}
	if = {
		limit = { has_realm_law = subroman_decay_2 }
		add_realm_law = subroman_decay_0
	}
	if = {
		limit = { has_realm_law = subroman_decay_3 }
		add_realm_law = subroman_decay_2
	}
	if = {
		limit = { has_realm_law = subroman_decay_4 }
		add_realm_law = subroman_decay_3
	}
}
#Replaces current clerical succession with new law
replace_current_clerical_succession = {
	if = {
		limit = { has_doctrine = doctrine_clerical_succession_temporal_appointment }
		remove_doctrine = doctrine_clerical_succession_temporal_appointment
	}
	else_if = {
		limit = { has_doctrine = doctrine_clerical_succession_spiritual_appointment }
		remove_doctrine = doctrine_clerical_succession_spiritual_appointment
	}
	else_if = {
		limit = { has_doctrine = doctrine_clerical_succession_temporal_fixed_appointment }
		remove_doctrine = doctrine_clerical_succession_temporal_fixed_appointment
	}
	else_if = {
		limit = { has_doctrine = doctrine_clerical_succession_spiritual_fixed_appointment }
		remove_doctrine = doctrine_clerical_succession_spiritual_fixed_appointment
	}
	else = { debug_log = "Found some unknown clerical succession doctrine. FIXME!" }
	add_doctrine = $THE_NEW_DOCTRINE$
}
