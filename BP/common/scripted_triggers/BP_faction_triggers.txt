#Liberty faction triggers

#NB: these trigger for having greater than 0 laws
realm_law_uses_crown_authority = { #Feudal/Clan
	OR = {
		has_realm_law = crown_authority_1
		has_realm_law = crown_authority_2
		has_realm_law = crown_authority_3
	}
}
realm_law_uses_high_tribal_authority = {
	OR = {

		has_realm_law = high_tribal_authority_3
	}
}

		has_realm_law = eranshar_authority_2
		has_realm_law = eranshar_authority_3
	}
}
#What realm laws support liberty factions
realm_law_supports_liberty_faction = {
	OR = {
		#Feudal/Clan Government
		realm_law_uses_crown_authority = yes
		#High Tribal Government
		realm_law_uses_high_tribal_authority = yes
		#Prefeudal Governments
		realm_law_uses_prefeudal_authority = yes
		#Gupta Governments
		realm_law_uses_gupta_authority = yes
		#Eranshar Governments
		realm_law_uses_eranshar_authority = yes
	}
}
