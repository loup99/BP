#General triggers
possesses_a_primary_titular_cultural_title = {
	custom_description = {
		text = has_a_valid_culture_title
		subject = root
		any_held_title = {
			is_titular = yes
			has_variable = title_culture
			save_temporary_scope_as = the_title
			OR = {
				root.culture = scope:the_title.var:title_culture
				AND = {
					root.culture = { any_parent_culture = { this = scope:the_title.var:title_culture } }
					root.culture = { has_same_culture_heritage = scope:the_title.var:title_culture }
				}
			}
		}
	}
}
general_kingdom_title_creation_trigger = {
	standard_claimant_status_trigger = yes
	possesses_a_primary_titular_cultural_title = no
	custom_description = {
		text = general_kingdom_title_creation_text
		subject = root.culture
		OR = {
			has_advanced_government_trigger = yes
			culture = { has_innovation = innovation_leges_barbarorum }
		}
	}
}
general_empire_title_creation_trigger = {
	standard_claimant_status_trigger = yes
	custom_description = { #Can't if I already have an empire
		text = already_have_an_empire
		NOT = { any_held_title = { tier = tier_empire } }
	}
	custom_description = { #Roman Empire cultures can only create Rome
		text = can_only_create_rome
		culture = { has_ability_to_reform_rome_directly_trigger = no }
	}
	trigger_if = { #Goths Gotta make Gothica if available
		limit = { culture = { the_culture_or_parent_is_with_heritage = { CULTURE = culture:gothic } } }
		custom_description = {
			text = goths_should_try_to_make_e_goths
			object = title:e_goths
			exists = title:e_goths.holder
		}
	}
	custom_description = {
		text = general_empire_title_creation_text
		subject = root.culture
		OR = {
			has_advanced_government_trigger = yes
			culture = { has_innovation = innovation_mundium }
		}
	}
}
