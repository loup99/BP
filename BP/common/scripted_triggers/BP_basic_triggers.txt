#Standardized title creation triggers
standard_claimant_status_trigger = {
	is_independent_ruler = yes
	is_adult = yes
	prestige_level >= $PRESTIGE_LEVEL$
}

#XOR Gate
XOR_trigger = {
	OR = {
		AND = {
			$A$
			NOT = { $B$ }
		}
		AND = {
			NOT = { $A$ }
			$B$
		}
	}
}
