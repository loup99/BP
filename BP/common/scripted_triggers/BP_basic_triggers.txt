valid_bureaucratic_government_state_trigger = {
	OR = {
		AND = {
			is_independent_ruler = yes
			primary_title.tier = tier_empire
		}
		AND = {
			is_independent_ruler = no
			any_liege_or_above = {
				has_government = bureaucratic_government
				primary_title.tier = tier_empire
			}
		}
	}
}

#recipient scope
can_demand_conversion = {
	#Zoroastrian approved cults can't forcibly convert each other
	trigger_if = {
		limit = { religion = religion:zoroastrianism_religion }
		NAND = {
			faith = { has_doctrine = special_doctrine_zoroastrian_approved_cult }
			$ACTOR$.faith = { has_doctrine = special_doctrine_zoroastrian_approved_cult }
		}
	}
	#Vanilla Condition
	trigger_else = { NOT = { faith = $ACTOR$.faith } }
}

#XOR Gate
XOR_trigger = {
	OR = {
		AND = {
			$A$
			NOT = { $B$ }
		}
		AND = {
			NOT = { $A$ }
			$B$
		}
	}
}
