can_get_military_command_trigger = {
	#Can only be under bureaucratic_government or military_command_government
	any_liege_or_above = {
		OR = {
			has_government = bureaucratic_government
			has_government = military_command_government
			has_government = subroman_government
		}
	}
	#Must be below emperor rank
	primary_title.tier <= tier_kingdom
	#Must not be independent
	is_independent_ruler = no
	#Can't be a bureaucratic_government
	NOT = { has_government = bureaucratic_government }
	#Not fully autonomous in imperial government
	NOT = { vassal_contract_has_flag = imperial_autonomy_full }
	#Can't be applied to empires ever
	NOT = { primary_title.tier = tier_empire }
}
