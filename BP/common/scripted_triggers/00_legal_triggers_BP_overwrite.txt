			can_change_succession_law_BP_trigger = yes
		}
		custom_description = {
			text = "no_powerful_vassal_with_negative_opinion"
			NOT = {
				any_vassal = {
					is_powerful_vassal = yes
					opposes_succession_law_change_trigger = yes
				}
			}
		}
	}
}
