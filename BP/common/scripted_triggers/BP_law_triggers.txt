#Vanilla Governments & sundry
realm_law_use_tribal_authority = {
	OR = {
		has_government = tribal_government
		has_government = nomad_government
	}
}
#BP Governments
realm_law_use_high_tribal_organization = { has_government = high_tribal_government }
realm_law_use_prefeudal_authority = { has_government = prefeudal_government }
realm_law_use_gupta_authority = { has_government = gupta_government }
realm_law_use_eranshar_authority = { has_government = eranshar_government }
realm_law_use_clerical_influence = {
	has_government = eranshar_government
	#Only the Shahanshah gets to deal with the influence of the priests on the government
	faith = {
		religion = religion:zoroastrianism_religion
		exists = this.religious_head
		this.religious_head = { is_vassal_of = root }
	}
}

realm_law_use_subroman_taxation = { has_government = subroman_government }
realm_law_use_feudal_transition = { has_government = subroman_government }

realm_law_use_bureaucratic_taxation = {
	OR = {
		has_government = bureaucratic_government
		has_government = gubernatorial_government
	}
}

#Title Law Triggers

#Only for bureaucratic_government
can_have_imperial_elective_succession_realm_law_trigger = { has_government = bureaucratic_government }

can_keep_imperial_elective_succession_law_trigger = {
	#The 'can_keep' triggers are dependent on actually having the law.
	#If they aren't, the trigger breakdown for the player breaks and shows incomplete information.
	trigger_if = { 
		limit = { has_realm_law = imperial_elective_succession_law }
	}
	can_have_imperial_elective_succession_realm_law_trigger = yes
}

can_have_feudal_elective_succession_title_law_trigger = {
	OR = {
		has_government = feudal_government
		has_government = prefeudal_government
		has_government = high_tribal_government
		has_government = subroman_government
	}
}
can_have_tribal_elective_succession_title_law_trigger = {
	OR = {
		has_government = tribal_government
		has_government = nomad_government
		has_government = high_tribal_government
	}
}
can_have_gaelic_elective_succession_title_law_trigger = {
	OR = {
		has_government = feudal_government
		has_government = clan_government
		has_government = tribal_government
		has_government = subroman_government
		has_government = high_tribal_government
	}
}

#### Crown Authority Equivalent levels
has_crown_authority_equivalent_level_trigger = {
	OR = {
		has_realm_law = crown_authority_$LEVEL$
		has_realm_law = high_tribal_organization_$LEVEL$
		has_realm_law = prefeudal_authority_$LEVEL$
		has_realm_law = gupta_authority_$LEVEL$
		has_realm_law = eranshar_authority_$LEVEL$
	}
}
