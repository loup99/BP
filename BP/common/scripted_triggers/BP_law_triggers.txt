can_have_feudal_elective_trigger = {
	has_government = feudal_government
	has_government = prefeudal_government
	has_government = gupta_government
	has_government = high_tribal_government
	has_government = subroman_government
	highest_held_title_tier >= tier_duchy
	NOR = { #Cultures have their special flavor.
		culture = culture:anglo_saxon
		has_culture_group = culture_group:north_germanic_group
	}
}

can_have_saxon_elective_succession_law_trigger = {
	OR = {
		has_government = feudal_government
		has_government = high_tribal_government
		has_government = tribal_government
	}
	highest_held_title_tier >= tier_kingdom
}
