	group = "special"
	
	is_available_on_create = {
		has_doctrine = special_doctrine_is_dualist_faith
	}

	special_doctrine_is_dualist_faith = {
		parameters = {
			hostility_override_tenet_gnosticism = 0
			hostility_override_special_doctrine_is_gnostic_faith = 0
			granting_titles_gives_stress = yes
		}
		traits = {
			virtues = { temperate }
			sins = { gluttonous }
		}
		character_modifier = {
			learning = 2
			stewardship = -2
		}
	}
}
