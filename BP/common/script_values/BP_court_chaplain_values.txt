conversion_holy_site_vaspurakan = {
	value = 0
	if = {
		limit = {
			scope:councillor.faith = {
				controls_holy_site = vaspurakan
			}
		}
		add = court_chaplain_conversion_base_total
		multiply = 0.50
	}
}
