#Negate de jure nerf
eranshar_negate_de_jure_nerf_multiplier = {
	value = 1
	if = {
		limit = {
			#If we're under e_persia
			scope:vassal = { eranshar_ignores_rightful_liege_trigger = yes }
			#And our liege is eranshar
			scope:liege = { has_government = eranshar_government }
			#and the liege is not a de jure liege, negate the effects of not being de jure
			NOR = {
				is_liege_a_direct_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
				is_liege_an_indirect_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
				is_liege_an_very_indirect_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
			}
		}
		multiply = 2.00 #Negate de jure penalty
	}
	else = { multiply = 1.00 }
}

#Levy calculation
eranshar_levies = {
	value = 0.25
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
