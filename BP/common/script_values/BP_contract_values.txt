#Negate de jure nerf
eranshar_negate_de_jure_nerf_multiplier = {
	value = 1
	if = {
		limit = {
			#If we're under e_persia
			scope:vassal = { eranshar_ignores_rightful_liege_trigger = yes }
			#And our liege is eranshar
			scope:liege = { has_government = eranshar_government }
			#and the liege is not a de jure liege, negate the effects of not being de jure
			NOR = {
				is_liege_a_direct_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
				is_liege_an_indirect_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
				is_liege_an_very_indirect_de_jure_holder = {
					LIEGE = scope:liege
					VASSAL = scope:vassal
				}
			}
		}
		multiply = 2.00 #Negate de jure penalty
	}
	else = { multiply = 1.00 }
}

#Eranshar Levies
#10-60%, based on opinion centered at 35%; players are -15 opinion equivalent
eranshar_government_levies_default = {
	value = 0.35
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_opinion_scale_positive = {
	value = 0.0025
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_opinion_scale_negative = {
	value = 0.0025
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_player = {
	value = 0.3125
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
#Equivalent to -75, -50, -25 opinions
eranshar_government_levies_min_ca_1 = {
	value = 0.1625
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_min_ca_2 = {
	value = 0.2250
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_min_ca_3 = {
	value = 0.2875
	multiply = eranshar_negate_de_jure_nerf_multiplier
}
eranshar_government_levies_max_possible = {
	value = eranshar_government_levies_default
	add = {
		value = eranshar_government_levies_opinion_scale_positive
		multiply = 100
	}
}
