### Eranshar revocation cost calculation
eranshar_title_revocation_count_counties = {
	value = 0
	scope:target = {
		every_in_de_jure_hierarchy = {
			limit = { this.tier = tier_county }
			add = 1
		}
	}
}
#I.e., how does crown authority effect this
eranshar_title_revocation_authority_value = {
	value = 1
	if = {
		limit = { scope:actor = { has_realm_law = eranshar_authority_3 } }
		multiply = 0.5
	}
	if = {
		limit = { scope:actor = { has_realm_law = eranshar_authority_2 } }
		multiply = 1.0
	}
	if = {
		limit = { scope:actor = { has_realm_law = eranshar_authority_1 } }
		multiply = 2.0
	}
	if = {
		limit = { scope:actor = { has_realm_law = eranshar_authority_0 } }
		multiply = 4.0
	}
}
