		limit = {
			exists = scope:county

			}

			value = scope:councillor.boost_county_conversion_phase_malus_value
			desc = CHAPLAIN_CONVERT_COUNTY_STRUGGLE_PENALTY
		}
	}

			exists = scope:county

				has_variable = learning_encouraged_conversion
			}

			desc = COURT_CHAPLAIN_CONVERSION_LEARNING_THEOLOGY
		}

			value = scope:councillor.conversion_minor_migration_minor_aristocracy_penalty

		}

		}

			value = scope:councillor.conversion_recent_emigration_bonus
			desc = COURT_CHAPLAIN_CONVERSION_RECENTLY_MIGRATED_AWAY
		}

	if = { # Have a short time to settle your migratory peoples

		}
