					value = -30

						desc = only_same_culture_acclaimation

				}

			if = {
				limit = {
					piety_level >= 1
					house ?= {
						OR = {
							has_house_power_parameter = candidate_score_per_piety_level_1
							has_house_power_parameter = candidate_score_per_piety_level_2
							has_house_power_parameter = candidate_score_per_piety_level_3

					value = 0

							limit = {
								house ?= { has_house_power_parameter = candidate_score_per_piety_level_3 }

							limit = {
								house ?= { has_house_power_parameter = candidate_score_per_piety_level_2 }
							}
							add = candidate_score_per_piety_level_2_value

							add = candidate_score_per_piety_level_1_value

				}

				limit = {
					scope:title.holder ?= {
