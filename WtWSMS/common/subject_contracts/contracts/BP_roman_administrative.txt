#Roman Empire's should be able to set things themselves at all levels
roman_administrative_themes = {
	display_mode = radiobutton
	is_shown = {
		scope:subject.primary_title.tier >= tier_duchy
	}
	obligation_levels = {
		roman_admin_theme_civilian = {
			is_shown = { scope:subject = { is_landed = yes } }
			default = yes
			position = { 0 0 }
			icon = "gfx/interface/icons/theme_administration_types/icon_civilian_administration.dds"
			gui_tags = { byzantine_purple }
			
			ai_liege_desire = {
				value = 0
				if = {
					limit = {
						scope:subject = {
							is_ai = yes
							OR = {
								has_trait = education_stewardship
								stewardship >= very_high_skill_rating
								AND = { # Large semi-coastal themes are good for an economic focus
									any_sub_realm_county = { count >= 2 is_coastal_county = yes }
									any_sub_realm_county = { count >= 2 is_coastal_county = no }
								}
							}
						}
					}
					add = 2
				}
				if = {
					limit = {
						scope:liege = {
							ai_has_builder_or_pious_builder_personality = yes
						}
					}
					add = 1
				}
			}
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.2 0.7 1.0 }

			tax_factor = 0.5
			levies_factor = -0.75

			subject_modifier = {
				development_growth_factor = 0.2
				build_gold_cost = -0.1
				build_speed = -0.1
			}
			
			enable_title_maa = no
			
			flag = admin_stewardship_obligation_bonus
			flag = admin_influence_construction_bonus
			flag = admin_ai_is_builder
			flag = admin_theme_civilian
			flag = obligation_high_taxes
		}
		roman_admin_theme_limitanei = {
			is_shown = { scope:subject = { is_landed = yes } }
			position = { 0 1 }
			icon = "gfx/interface/icons/theme_administration_types/icon_frontier_administration.dds"
			gui_tags = { byzantine_purple }
			
			ai_liege_desire = {
				value = 0
				if = {
					limit = {
						scope:subject = {
							is_ai = yes
						}
					}
					add = 5
				}
			}
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.7 0.2 1.0 }
			
			is_valid = {
				custom_tooltip = {
					text = roman_admin_theme_frontier_valid_desc
					scope:subject = {
						any_sub_realm_county = {
							any_neighboring_county = {
								holder = {
									trigger_if = {
										limit = { exists = this.suzerain }
										suzerain != scope:subject.top_liege
									}
									trigger_else = {top_liege != scope:subject.top_liege }
									top_liege = { government_has_flag = government_is_tribal }
									top_liege = { highest_held_title_tier >= tier_duchy }
								}
							}
						}
					}
				}
			}
			
			tax_factor = -0.5

			subject_modifier = {
				fort_level = 1
				defender_advantage = 3
				hostile_county_attrition = -0.2
				hostile_raid_time = 0.50
				
				monthly_treasury_from_military_budget_base = 1.5
			}
			
			flag = admin_prowess_obligation_bonus
			flag = admin_ai_is_warlike
			flag = admin_theme_frontier
			flag = admin_theme_can_raid
		}
		roman_admin_theme_naval = {
			is_shown = { scope:subject = { is_landed = yes } }
			position = { 0 2 }
			icon = "gfx/interface/icons/theme_administration_types/icon_naval_administration.dds"
			gui_tags = { byzantine_purple }
			
			ai_liege_desire = {
				value = 0
				if = {
					limit = {
						scope:subject = { is_ai = yes }
					}
					if = {
						limit = { # Large coastal areas qualify as good naval themes
							scope:subject = {
								any_sub_realm_county = {
									count >= 3
									is_coastal_county = yes
								}
							}
							scope:liege = { # But don't go overboard with naval themes
								any_vassal = {
									percent < 0.15
									vassal_contract_has_flag = admin_theme_naval
								}
							}
						}
						add = 3
					}
					
					if = {
						limit = { # Islands are good naval candidates
							scope:subject = {
								any_sub_realm_county = {
									percent >= 0.95
									is_coastal_county = yes
									OR = {
										any_neighboring_county = {
											percent >= 0.95
											holder = {
												OR = {
													this = scope:subject
													any_liege_or_above = { this = scope:subject }
												}
											}
										}
										any_neighboring_county = { count < 1 }
									}
								}
							}
						}
						add = 8
					}
				}
			}
			ai_subject_desire = 0
			
			score = 0
			color = { 0.0 0.4 0.7 1.0 }
			
			is_valid = {
				custom_tooltip = {
					text = admin_theme_naval_valid_desc
					scope:subject = {
						any_sub_realm_county = {
							is_coastal_county = yes
						}
					}
				}
				scope:liege = {
					highest_held_title_tier >= tier_empire
				}
				custom_tooltip = {
					text = roman_admin_theme_naval_valid_desc
					OR = {
						scope:liege = {
							NOT = {
								any_vassal = { vassal_contract_has_flag = admin_theme_naval }
							}
						}
						AND = {
							scope:subject = { vassal_contract_has_flag = admin_theme_naval }
							scope:liege = {
								any_vassal = {
									count = 1
									vassal_contract_has_flag = admin_theme_naval
								}
							}
						}
					}
				}
			}
			
			liege_modifier = {
				embarkation_cost_mult = -0.1
				naval_movement_speed_mult = 0.1
			}

			subject_modifier = {
				embarkation_cost_mult = -0.5
				naval_movement_speed_mult = 0.25
				no_disembark_penalty = yes
				coastal_advantage = 10
				
				monthly_treasury_from_military_budget_base = 1
			}
			
			enable_title_maa = no
			
			flag = admin_tradeport_obligation_bonus
			flag = admin_theme_naval
		}
		#Special "hidden" themes for landless rulers as this seems to break things...
		roman_admin_theme_ministry_civilian = {
			position = { 1 0 }
			icon = "gfx/interface/icons/theme_administration_types/icon_civilian_administration.dds"
			gui_tags = { byzantine_purple }
			
			is_shown = { scope:subject = { is_landed = no } }
			is_valid = {
				custom_tooltip = {
					text = available_only_for_ministers
					scope:subject = { is_landless_ruler = yes }
					scope:subject = { any_held_title = { count <= 0 has_variable = is_supreme_roman_military_title } }
				}
			}
			
			ai_liege_desire = 0
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.7 0.2 1.0 }
			
			enable_title_maa = no

			subject_modifier = {}
		}
		roman_admin_theme_ministry_military = {
			position = { 1 0 }
			icon = "gfx/interface/icons/theme_administration_types/icon_frontier_administration.dds"
			gui_tags = { byzantine_purple }
			
			is_shown = { scope:subject = { is_landed = no } }
			is_valid = {
				scope:subject = { is_landless_ruler = yes }
				scope:subject = { any_held_title = { has_variable = is_supreme_roman_military_title } }
			}
			
			ai_liege_desire = 0
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.7 0.2 1.0 }
			
			enable_title_maa = yes

			subject_modifier = {
				men_at_arms_title_cap = 5
				men_at_arms_title_limit = 5
				
				monthly_treasury_from_military_budget_mult = 2.5
			}
		}
	}
}

roman_administrative_ministry_obligations = {
	display_mode = tree
	defaults_to_highest_valid_level = yes
	is_shown = {
		scope:subject = {
			OR = {
				any_held_title = { has_variable = is_supreme_roman_military_title }
				any_held_title = { has_variable = is_supreme_roman_ministry_title }
			}
		}
	}
	can_be_changed = {
		scope:subject = {
			trigger_if = {
				limit = {
					any_held_title = { has_variable = is_supreme_roman_military_title }
				}
				NOT = { vassal_contract_has_flag = roman_administrative_marshal_obligation }
			}
			trigger_else = {
				always = no
			}
		}
	}
	obligation_levels = {
		roman_administrative_minister_none = {

				value = 1
			}
			ai_subject_desire = 0

			score = 1
			color = { 0.7 0.7 0.7 1.0 }

			subject_modifier = {
			}

			flag = roman_administrative_minister_obligation
		}
		roman_administrative_minister_civil = {
			position = { 1 0 }
			icon = "gfx/interface/icons/celestial_administration_types/icon_standard_administration.dds"
			gui_tags = { civilian }
			parent = roman_administrative_minister_none

			ai_liege_desire = {
				value = 2
			}
			ai_subject_desire = 0
			
			is_valid = {
				NOT = { scope:subject = { any_held_title = { has_variable = is_supreme_roman_military_title } } }
			}

			score = 2
			color = { 0.7 0.7 0.7 1.0 }

			enable_title_maa = no

		}

			position = { 2 0 }
			icon = "gfx/interface/icons/celestial_administration_types/icon_military_administration.dds"
			gui_tags = { military }

			}
			ai_subject_desire = 0

			}

			}
