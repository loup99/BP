					limit = {
						scope:subject = {
							is_ai = yes
							OR = {
								has_trait = education_stewardship
								stewardship >= very_high_skill_rating

									any_sub_realm_county = { count >= 2 is_coastal_county = yes }

								}
							}
						}
					}
					add = 2
				}

					limit = {
						scope:liege = {
							ai_has_builder_or_pious_builder_personality = yes
						}
					}

				}
			}
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.2 0.7 1.0 }

			levies_factor = -0.75

				development_growth_factor = 0.2

			flag = admin_influence_construction_bonus
			flag = admin_ai_is_builder

			flag = obligation_high_taxes
		}

			is_shown = { scope:subject = { is_landed = yes } }
			position = { 1 0 }

			gui_tags = { byzantine_purple }
			
			ai_liege_desire = {
				value = 0
				if = {
					limit = {
						scope:subject = {
							is_ai = yes
						}
					}
					add = 5
				}
			}
			ai_subject_desire = 0
			
			score = 0
			color = { 0.2 0.7 0.2 1.0 }
			
			is_valid = {
				custom_tooltip = {
					text = roman_admin_theme_frontier_valid_desc
					scope:subject = {
						any_sub_realm_county = {
							any_neighboring_county = {
								holder = {
									trigger_if = {
										limit = { exists = this.suzerain }
										suzerain != scope:subject.top_liege
									}
									trigger_else = {top_liege != scope:subject.top_liege }
									top_liege = { government_has_flag = government_is_tribal }
									top_liege = { highest_held_title_tier >= tier_duchy }
								}
							}
						}
					}
				}
			}
			
			tax_factor = -0.5

			subject_modifier = {
				fort_level = 1
				defender_advantage = 3
				hostile_county_attrition = -0.2

				monthly_treasury_from_military_budget_base = 1.5
			}
			
			flag = admin_prowess_obligation_bonus
			flag = admin_ai_is_warlike
			flag = admin_theme_frontier
			flag = admin_theme_can_raid
		}
		roman_admin_theme_naval = {
			is_shown = { scope:subject = { is_landed = yes } }
			position = { 2 0 }
			icon = "gfx/interface/icons/theme_administration_types/icon_naval_administration.dds"
			gui_tags = { byzantine_purple }
			
			ai_liege_desire = {
				value = 0
				if = {
					limit = {
						scope:subject = { is_ai = yes }
					}
					if = {
						limit = { # Large coastal areas qualify as good naval themes
							scope:subject = {
								any_sub_realm_county = {
									count >= 3
									is_coastal_county = yes
								}
							}
							scope:liege = { # But don't go overboard with naval themes
								any_vassal = {
									percent < 0.15
									vassal_contract_has_flag = admin_theme_naval
								}
							}
						}
						add = 3
					}
					
					if = {
						limit = { # Islands are good naval candidates
							scope:subject = {
								any_sub_realm_county = {
									percent >= 0.95
									is_coastal_county = yes
									OR = {
										any_neighboring_county = {
											percent >= 0.95
											holder = {
												OR = {
													this = scope:subject
													any_liege_or_above = { this = scope:subject }
												}
											}
										}
										any_neighboring_county = { count < 1 }
									}
								}
							}
						}
						add = 8
					}
				}
			}

			score = 0
			color = { 0.0 0.4 0.7 1.0 }

				custom_tooltip = {

				}

					text = roman_admin_theme_naval_valid_desc
					OR = {
						scope:liege = {
							NOT = {
								any_vassal = { vassal_contract_has_flag = admin_theme_naval }
							}
						}
						AND = {
							scope:subject = { vassal_contract_has_flag = admin_theme_naval }
							scope:liege = {
								any_vassal = {
									count = 1
									vassal_contract_has_flag = admin_theme_naval
								}
							}
						}
					}
				}
			}
			
			liege_modifier = {
				embarkation_cost_mult = -0.1
				naval_movement_speed_mult = 0.1
			}

				embarkation_cost_mult = -0.5
				naval_movement_speed_mult = 0.25
				no_disembark_penalty = yes
				coastal_advantage = 10
				
				monthly_treasury_from_military_budget_base = 1
			}
			
			enable_title_maa = no
			
			flag = admin_tradeport_obligation_bonus
			flag = admin_theme_naval
		}
		##### Sub-Roman Themes
		roman_admin_theme_military = {
			#This is for land 
			is_shown = {
				scope:liege = { government_has_flag = government_is_subroman_administrative }
				scope:subject = { government_has_flag = government_is_subroman_administrative }
				scope:subject = { is_landed = yes }
			}
			is_valid = {
				scope:liege = { government_has_flag = government_is_subroman_administrative }
				scope:subject = { government_has_flag = government_is_subroman_administrative }

				trigger_if = {
					limit = { NOT = { scope:subject = { vassal_contract_has_flag = roman_landed_military_contract } } }
					custom_description = {
						text = has_not_reached_military_province_limits
						subject = scope:subject.top_liege
						scope:subject.top_liege.number_of_military_provinces < scope:subject.top_liege.number_of_military_provinces_allowed
					}
				}
				trigger_if = {
					limit = { NOT = { scope:subject.top_liege.culture = { has_cultural_parameter = greco_roman_social_tradition } } }
					custom_description = {
						text = non_roman_cultures_may_make_only_certain_holders_military_provinces

				monthly_treasury_from_military_budget_base = 4.0
			}

			enable_title_maa = yes

		##### Hidden ministry level-theme to avoid title MAA issues
		roman_admin_theme_ministry = {
