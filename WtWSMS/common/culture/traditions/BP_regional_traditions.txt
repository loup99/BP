tradition_marauders = {
	category = combat
	
	layers = {
		0 = martial
		1 = mediterranean
		4 = battle.dds
	}
	
	is_shown = { always = no }
	can_pick = {
		traditions_body_status_type_is_in_list_trigger = no
		traditions_cultural_combat_type_is_in_list_trigger = no
		custom_tooltip = {
			text = cannot_have_tradition_practiced_pirates
			NOT = { any_in_list = { list = traits this = flag:tradition_practiced_pirates } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_battlefield_looters
			NOT = { any_in_list = { list = traits this = flag:tradition_battlefield_looters } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_staunch_traditionalists
			NOT = { any_in_list = { list = traits this = flag:tradition_staunch_traditionalists } }
		}
		traditions_body_status_type_is_in_list_trigger = no
	}
	can_pick_for_hybridization = {
		traditions_body_status_type_is_in_list_trigger = no
		traditions_cultural_combat_type_is_in_list_trigger = no
		custom_tooltip = {
			text = cannot_have_tradition_practiced_pirates
			NOT = { any_in_list = { list = traits this = flag:tradition_practiced_pirates } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_battlefield_looters
			NOT = { any_in_list = { list = traits this = flag:tradition_battlefield_looters } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_staunch_traditionalists
			NOT = { any_in_list = { list = traits this = flag:tradition_staunch_traditionalists } }
		}
		traditions_body_status_type_is_in_list_trigger = no
		custom_description = {
			text = hybrid_must_have_anatolian_heritage_for_tradition_marauders
			any_in_list = { list = traits this = flag:heritage_anatolian }
		}
	}
	
	culture_modifier = {
		cultural_acceptance_gain_mult = -1.00
	}
	parameters = {
		culture_can_raid_even_if_feudal = yes
		not_tribal_raiding_malus = yes
		physical_disfigurement_blocks_inheritance = yes
		expensive_convert_to_local_culture = yes
		harder_to_hybridize = yes
		harder_to_convert_county_culture_without_heritage = yes
		populist_factions_more_common_against_foreigners = yes
		disloyal_trait_more_common = yes
	}
	character_modifier = {
		#Settled, non-theocratic governments consider you hostile
		feudal_government_vassal_opinion = -20
		republic_government_vassal_opinion = -20
		prefeudal_government_vassal_opinion = -20
		gupta_government_vassal_opinion = -20
		eranshar_government_vassal_opinion = -20
		subroman_government_vassal_opinion = -20
		gubernatorial_government_vassal_opinion = -20
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						any_in_list = { list = traits this = flag:ethos_bellicose }
						any_in_list = { list = traits this = flag:ethos_spiritual }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_spiritual_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = { NOT = { any_in_list = { list = traits this = flag:ethos_bellicose } } }
			multiply = 0.5
		}
	}
}

tradition_asturian_hierarchy = {
	category = regional

	layers = {
		0 = intrigue
		1 = mediterranean
		4 = crown.dds
	}
	
	is_shown = { always = no }
	can_pick = { the_culture_or_parent_or_above_is_trigger = { CULTURE = culture:astures } }

	parameters = {
		is_a_heritage_tradition = yes
		not_male_succession = yes
		female_only_inheritance = yes
		female_preference_inheritance = yes
	}
	
	character_modifier = {
		opinion_of_female_rulers = 10
		opinion_of_male_rulers = -10
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						has_game_rule = inversed_gender_equality
						any_in_list = { list = traits this = flag:ethos_egalitarian }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_egalitarian_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = { scope:character = { is_ai = no } }
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								is_female = yes
							}
						}
					}
					trigger_else = { always = no }
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = five_female_rulers_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 25
		if = {
			limit = { NOT = { scope:character = { is_female = yes } } }
			multiply = 0
		}
	}
}
