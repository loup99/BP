tradition_marauders = {
	category = combat
	
	layers = {
		0 = martial
		1 = mediterranean
		4 = battle.dds
	}
	
	is_shown = { always = no }
	can_pick = {
		traditions_body_status_type_is_in_list_trigger = { TRADITION = culture_tradition:tradition_marauders }
		traditions_cultural_combat_type_is_in_list_trigger = { TRADITION = culture_tradition:tradition_marauders }

			text = cannot_have_tradition_practiced_pirates
			NOT = { culture_tradition:tradition_practiced_pirates = { is_in_list = traits } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_battlefield_looters
			NOT = { culture_tradition:tradition_battlefield_looters = { is_in_list = traits } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_staunch_traditionalists
			NOT = { culture_tradition:tradition_staunch_traditionalists = { is_in_list = traits } }
		}
	}
	can_pick_for_hybridization = {
		traditions_body_status_type_is_in_list_trigger = { TRADITION = culture_tradition:tradition_marauders }
		traditions_cultural_combat_type_is_in_list_trigger = { TRADITION = culture_tradition:tradition_marauders }
		custom_tooltip = {
			text = cannot_have_tradition_practiced_pirates
			NOT = { culture_tradition:tradition_practiced_pirates = { is_in_list = traits } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_battlefield_looters
			NOT = { culture_tradition:tradition_battlefield_looters = { is_in_list = traits } }
		}
		custom_tooltip = {
			text = cannot_have_tradition_staunch_traditionalists
			NOT = { culture_tradition:tradition_staunch_traditionalists = { is_in_list = traits } }
		}
		custom_description = {
			text = hybrid_must_have_anatolian_heritage_for_tradition_marauders
			NOT = { culture_pillar:heritage_anatolian = { is_in_list = traits } }
		}
	}
	
	culture_modifier = {
		cultural_acceptance_gain_mult = -1.00
	}
	parameters = {
		culture_can_raid_over_land_even_if_feudal = yes
		not_tribal_raiding_malus = yes
		physical_disfigurement_blocks_inheritance = yes
		expensive_convert_to_local_culture = yes
		harder_to_hybridize = yes
		harder_to_convert_county_culture_without_heritage = yes
		populist_factions_more_common_against_foreigners = yes
		disloyal_trait_more_common = yes
	}
	character_modifier = {
		direct_vassal_opinion = -20
		courtly_same_culture_opinion = 20
		glory_hound_same_culture_opinion = 20
		parochial_same_culture_opinion = 20
		zealot_same_culture_opinion = 20
		minority_same_culture_opinion = 20
		barons_and_minor_landholders_same_culture_opinion = 20
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_spiritual = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_spiritual_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = { NOT = { culture_pillar:ethos_bellicose = { is_in_list = traits } } }
			multiply = 0.5
		}
	}
}

#Berber Traditions
tradition_foggara_builders = { #Garamantes civilization
#[1] - Referencing the fact that the Garamantes seem to have been agriculturalists primarily
	category = realm

	layers = {
		0 = steward
		1 = mena
		4 = water_technology.dds
	}

	is_shown = {
		# DLC check.
		always = no
		NOT = { has_cultural_tradition = tradition_dryland_dwellers }
	}
	
	can_pick = {
		custom_description = {
			text = culture_in_dryland_desc
			any_culture_county = {
				any_county_province = {
					OR = {
						terrain = drylands
						terrain = desert
						terrain = desert_mountains
					}
				}
			}
		}
		# Shouldn't be combined with the following traditions:
		custom_tooltip = {
			text = cannot_have_tradition_dryland_dwellers
			NOT = { culture_tradition:tradition_dryland_dwellers = { is_in_list = traits } } #this is another approach to drylands dwelling
		}
		custom_tooltip = {
			text = cannot_have_tradition_saharan_nomads
			NOT = { culture_tradition:tradition_saharan_nomads = { is_in_list = traits } } #this is another approach to desert life
		}
	}
	
	parameters = {
		martial_education_worse_outcomes = yes #[1]
		plantations_building_bonuses = yes
		drylands_hunt_success_chance = yes
		unlocks_desert_buildings_early = yes
	}
	character_modifier = {
		men_at_arms_recruitment_cost = 0.25 #[1]
		prowess = -2 #[1]
	}
	county_modifier = {
		oasis_development_growth_factor = 0.10
		desert_development_growth_factor = 0.05
	}
	province_modifier = {
		oasis_construction_gold_cost = -0.10
		oasis_holding_construction_gold_cost = -0.10
		desert_construction_gold_cost = -0.05
		desert_holding_construction_gold_cost = -0.05
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bureaucratic = { is_in_list = traits }
						culture_pillar:ethos_stoic = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_stoic_or_bureaucratic_desc
				}
			}
			if = {
				limit = {
					NOT = {
						any_culture_county = {
							percent >= 0.50
							any_county_province = {
								OR = {
									terrain = oasis
									terrain = desert
								}
							}
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = dry_percentage_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOT = { #if the realm isn't very dry, the AI won't pick this
					scope:character = {
						any_sub_realm_county = {
							percent >= 0.5
							culture = scope:character.culture
							any_county_province = {
								OR = {
									terrain = oasis
									terrain = desert
								}
							}
						}
					}
				}
			}
			multiply = 0
		}
	}
}

	category = regional

	layers = {
		0 = intrigue
		1 = mediterranean
		4 = crown.dds
	}
	
	is_shown = { always = no }
	can_pick = { the_culture_or_parent_or_above_is_trigger = { CULTURE = culture:astures } }

	parameters = {
		is_a_legal_tradition = yes
		not_male_succession = yes
		female_only_inheritance = yes
		female_preference_inheritance = yes
	}
	
	character_modifier = {
		opinion_of_female_rulers = 10
		opinion_of_male_rulers = -10
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						has_game_rule = inversed_gender_equality
						culture_pillar:ethos_egalitarian = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_egalitarian_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = { scope:character = { is_ai = no } }
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								is_female = yes
							}
						}
					}
					trigger_else = { always = no }
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = five_female_rulers_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 25
		if = {
			limit = { NOT = { scope:character = { is_female = yes } } }
			multiply = 0
		}
	}
}
