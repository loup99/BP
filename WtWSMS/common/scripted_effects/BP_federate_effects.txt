change_government_roman_federate_effect = {
	change_government = roman_federate_government
	if = {
		limit = {
			this = this.house.house_head
			this.primary_title.tier >= tier_duchy #Don't want nf running counties...
			any_held_title = { count < 1 is_noble_family_title = yes }
		}
		give_noble_family_title = {
			name = noble_family_name
			tier = duchy
			article = DEFAULT_TITLE_NAME_ARTICLE
			government = roman_federate_government
			save_scope_as = new_title
		}
	}
}
