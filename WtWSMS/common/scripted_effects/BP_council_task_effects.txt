apply_minority_faith_promotion_effect = { #Only to be used in common/00_court_chaplain_tasks_BP_overwrite.txt (multiple scopes)
	scope:county = {
		if = {
			limit = { the_county_faith_is_what_is_expected_trigger = yes }
			demote_random_faith_minority_effect = yes
		}
		else = {
			if = {
				limit = { the_county_should_convert_to_hof_faith_trigger = yes } #Special Zoro thing
				promote_faith_minority_effect = { FAITH = $TARGET$.faith.religious_head.faith }
			}
			else = { promote_faith_minority_effect = { FAITH = $TARGET$.faith } }
			if = {
				limit = { the_county_faith_is_what_is_expected_trigger = yes }
				scope:councillor = { start_default_task = yes }
			}
		}
	}
}
