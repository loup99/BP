bp_propagate_ecumenism_effect = {
	#Christian Faiths
	if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_nicene_ecumenism } }
		add_doctrine = special_doctrine_nicene_ecumenism
	}
	else_if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_alexandria_ecumenism } }
		add_doctrine = special_doctrine_alexandria_ecumenism
	}
	else_if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_nestorian_ecumenism } }
		add_doctrine = special_doctrine_nestorian_ecumenism
	}
	else_if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_nestorian_ecumenism } }
		add_doctrine = special_doctrine_nestorian_ecumenism
	}
	#Legalized Faiths under Persian Aegis
	else_if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_arian_ecumenism } }
		add_doctrine = special_doctrine_arian_ecumenism
	}
	#Zoroastrian Faiths
	else_if = {
		limit = { scope:old_faith = { has_doctrine = special_doctrine_zoroastrian_approved_cult } }
		add_doctrine = special_doctrine_zoroastrian_approved_cult
	}
}
#character scope
bp_specific_court_language_effect = {
	if = {
		limit = {
			has_royal_court = yes
			scope:title.tier >= tier_kingdom
		}
		if = { #Bureaucratic Case
			limit = { has_government = bureaucratic_government }
			if = {
				limit = { $CHAR$.primary_title = { has_variable = bureaucratic_language } }
				set_bureaucratic_court_language_effect = {
					TITLE = $CHAR$.primary_title
					CHARACTER = $CHAR$
				}
			}
		}
		else_if = { #Gubernatorial Case
			limit = { has_government = gubernatorial_government }
			if = {
				limit = { $CHAR$.top_liege.primary_title = { has_variable = bureaucratic_language } }
				set_bureaucratic_court_language_effect = {
					TITLE = $CHAR$.top_liege.primary_title
					CHARACTER = $CHAR$
				}
			}
		}
		else_if = { #Mandala Case
			limit = { has_government = mandala_government }
			set_court_language = language_sanskrit
		}
	}
}

#Mostly for our game-start on action stuff
change_faith_if_alive = {
	if = {
		limit = { exists = $CHARACTER$ }
		$CHARACTER$ = {
			if = {
				limit = { is_alive = yes }
				set_character_faith_history = $FAITH$
			}
		}
	}
}
