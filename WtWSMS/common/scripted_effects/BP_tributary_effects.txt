update_federate_modifiers_effect = {
	if = {
		limit = { has_character_modifier = federate_federate_subsidy_modifier }
		remove_character_modifier = federate_federate_subsidy_modifier
	}
	add_character_modifier = federate_federate_subsidy_modifier
	if = {
		limit = { has_character_modifier = federate_federate_levy_modifier }
		remove_character_modifier = federate_federate_levy_modifier
	}
	add_character_modifier = federate_federate_levy_modifier
}
