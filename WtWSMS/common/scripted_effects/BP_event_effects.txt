reform_caucasian_albania_effect = {

	}

form_frankish_kingdom_effect = {
	#If you have a titular Frankish Duchy, replace the COA with your kingdom title
	if = {
		limit = { primary_title = title:d_salia }
		title:k_franks = { set_coa = title:d_salia }
	}
	else_if = {
		limit = { primary_title = title:d_mosania }
		title:k_franks = { set_coa = title:d_mosania }
	}
	else_if = {
		limit = { primary_title = title:d_baroia }
		title:k_franks = { set_coa = title:d_baroia }
	}
	else_if = {
		limit = { primary_title = title:d_ripuria }
		title:k_franks = { set_coa = title:d_ripuria }
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	title:k_franks = {
		change_title_holder = {
			holder = root
			change = scope:change
		}
	}
	#Shift the capital Duchy, always
	this.capital_county.duchy = { set_de_jure_liege_title = title:k_franks }
	#Shift every duchy that's completely controlled
	every_realm_county = { this.duchy = { add_to_list = possible_duchies } }
	every_in_list = { #If it's completely controlled and neighboring the capital, shift its de jure liege
		list = possible_duchies
		limit = {
			root = { completely_controls = prev.duchy }
			any_county = {
				this.duchy = prev
				any_neighboring_county = { this.duchy = root.capital_county.duchy }
			}
		}
		set_de_jure_liege_title = title:k_franks
	}
	resolve_title_and_vassal_change = scope:change
	
	#Make the kingdom of the Franks part of some de jure empire
	hidden_effect = {
		title:k_franks = { set_de_jure_liege_title = title:e_gaul }
	}
	
	#Move capital to Aachen if its not held by a human player
	if = {
		limit = {
			any_realm_county = { this = title:c_aachen }
			title:c_aachen.holder = { is_ai = yes }
		}
		create_title_and_vassal_change = {
			type = returned
			save_scope_as = change
			add_claim_on_loss = no
		}
		title:c_aachen = {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change
		set_realm_capital = title:c_aachen
	}
	else_if = {
		limit = { title:c_aachen.holder = root }
		set_realm_capital = title:c_aachen
	}
	
	#Give the special Soissons subjugation CB
	custom_description = {
		text = unlocks_one_major_cb_against_roman_state
		add_character_flag = available_realm_conquest_cb_against_centralized_state
	}
	
	#Add claims to Gaul
	add_pressed_claim = title:e_gaul
	if = {
		limit = { culture = { has_cultural_parameter = male_only_inheritance } }
		every_heir = {
			limit = { is_female = no }
			add_pressed_claim = title:e_gaul
		}
	}
	else_if = {
		limit = { culture = { has_cultural_parameter = female_only_inheritance } }
		every_heir = {
			limit = { is_male = no }
			add_pressed_claim = title:e_gaul
		}
	}
	else = { every_heir = { add_pressed_claim = title:e_gaul } }
}

	legalize_caucasian_christianity_global_effect = yes
	add_piety_level = -1
	#The clerics are angry
	every_vassal_or_below = {
		limit = {
			religion = religion:zoroastrianism_religion
			faith = { has_doctrine = doctrine_major_branch_behdin }
		}
		if = {
			limit = { government_has_flag = government_is_theocracy }
			add_opinion = {
				modifier = angry_opinion
				target = root
				opinion = -20
			}
		}
	}
	custom_description = {
		text = every_court_chaplain_is_angry
		every_vassal_or_below = {
			limit = {
				religion = religion:zoroastrianism_religion
				faith = { has_doctrine = doctrine_major_branch_behdin }
			}
			every_councillor = {
				limit = { has_council_position = councillor_court_chaplain }
				add_opinion = {
					modifier = angry_opinion
					target = root
					opinion = -20
				}
			}
		}
	}
	#All the Christians in the Empire are relieved
