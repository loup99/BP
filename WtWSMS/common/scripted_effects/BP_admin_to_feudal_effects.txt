improve_aristocracy_law_one_level_effect = {
	if = {
		limit = { has_realm_law = subroman_aristocracy_law_0 }
		add_realm_law = subroman_aristocracy_law_1
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_1 }
		add_realm_law = subroman_aristocracy_law_2
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_2 }
		add_realm_law = subroman_aristocracy_law_3
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_3 }
		add_realm_law = subroman_aristocracy_law_4
	}
}
decline_aristocracy_law_one_level_effect = {
	if = {
		limit = { has_realm_law = subroman_aristocracy_law_1 }
		add_realm_law = subroman_aristocracy_law_0
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_2 }
		add_realm_law = subroman_aristocracy_law_1
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_3 }
		add_realm_law = subroman_aristocracy_law_2
	}
	else_if = {
		limit = { has_realm_law = subroman_aristocracy_law_4 }
		add_realm_law = subroman_aristocracy_law_3
	}
}
