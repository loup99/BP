make_the_tributaries_effect = {
	#Make Byzantium the Suzarein
	create_title_and_vassal_change = {
		type = swear_fealty
		save_scope_as = make_suzarein
		add_claim_on_loss = no
	}
	every_in_list = {
		list = $THE_LIST_OF_TRIBS$
		this.holder = {
			change_liege = {
				liege = $LIEGE$
				change = scope:make_suzarein
			}
		}
	}
	resolve_title_and_vassal_change = scope:make_suzarein
	#Apply the Special Vassal Contract
	every_in_list = {
		list = $THE_LIST_OF_TRIBS$
		this.holder = {
			vassal_contract_set_obligation_level = { type = tribute_contract level = 1 }
			set_vassal_contract_modification_blocked = yes
		}
	}
}
