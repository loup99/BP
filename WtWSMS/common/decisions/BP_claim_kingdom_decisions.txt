##### These are various decisions that cover weird edge cases between the CK3 mechanics
##### and history, like the throne of Aghwank being empty for a few decades then being reclaimed.
#####
##### Basically, should be a cheaper, simpler way of getting certain kingdoms that would exist
##### exist historically
reclaim_albania_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}
	decision_group_type = major
	ai_check_interval = 60
	desc = reclaim_albania_decision_desc
	selection_tooltip = reclaim_albania_decision_tooltip
	
	is_shown = {
		highest_held_title_tier = tier_duchy
		OR = {
			has_title = title:d_caucasian_albania
			has_title = title:d_balasagan
		}
		has_claim_on = title:k_caucasian_albania
		NOT = { has_global_variable = reestablished_albania }
	}
	
	is_valid = {
		OR = {
			AND = {
				legitimacy_level >= 3
				prestige_level >=3
			}
			has_trait = legendary_leader
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_at_peace_adult = yes
		is_imprisoned = no
		is_adult = yes
	}
	
	cost = {
		gold = 150
		prestige = 200
	}
	
	effect = {
		#Done as event to prevent RC formation event from firing
		show_as_tooltip = { reform_caucasian_albania_effect = yes }
		hidden_effect = {
			#Can only do it once
			set_global_variable = {
				name = reestablished_albania
				value = yes
			}
			
			root = { save_scope_as = the_albanian_reformer }
			
			root = { trigger_event = { id = BP_claim_kingdom_decisions.0001 } }
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 100
	}
}
#The Aures Kingdom in the mountains of Africa

		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}
	decision_group_type = major
	ai_check_interval = 60
	desc = claim_k_aures_decision_desc
	selection_tooltip = claim_k_aures_decision_tooltip

		highest_held_title_tier = tier_duchy
		has_title = title:d_auresia
		NOT = { has_global_variable = claimed_k_aures }
	}
	
	is_valid = {
		OR = {
			#Has enough political clout
			prestige_level >= 2
			piety_level >= 2
		}
		has_title = title:d_auresia
		#Have to be in the Aures Mountains
		OR = {
			custom_description = {
				text = capital_is_bilizma
				root.capital_county = title:c_bilizma
			}
			custom_description = {
				text = capital_is_biskra
				root.capital_county = title:c_biskra
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_at_peace_adult = yes
		is_imprisoned = no

	}
	
	cost = {
		gold = 25
		prestige = 50
		piety = 50
	}
	
	effect = {
		#Done as event to prevent RC formation event from firing
		show_as_tooltip = { form_k_aures_effect = yes }

	ai_will_do = { base = 100 }

#The Frankish Kingdom
form_frankish_kingdom_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/pharamond_on_a_shield.dds"
	}
	decision_group_type = major
	ai_check_interval = 60
	desc = form_frankish_kingdom_decision_desc
	selection_tooltip = form_frankish_kingdom_decision_tooltip
	
	is_shown = {
		culture = culture:old_frankish
		NOT = { exists = title:k_franks.holder }
		NOT = { has_global_variable = formed_frankish_kingdom }
	}
	
	is_valid = {
		is_independent_ruler = yes
		#Control at least two capitals
		any_realm_county = {
			count >= 2
			custom_description = {
				text = list_frank_capitals
				OR = {
					this = title:c_hainaut
					this = title:c_cambray
					this = title:c_cologne
					this = title:c_frankfurt
				}
			}
		}
		#Must control 66% of your culture's counties
		AND = {
			custom_description = {
				text = controls_least_two_thirds_of_counties_with_culture
				subject = root
				check_if_control_percentage_of_all_counties_with_condition_trigger = {
					FEATURE = culture
					CHARACTER = root
					PERCENTAGE = 0.66
				}
			}
		}
		OR = {
			AND = {
				prestige_level >= 3
				legitimacy_level >= 3
			}
			has_trait = legendary_leader
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_at_peace_adult = yes
		is_imprisoned = no
		is_adult = yes
	}
	
	cost = {
		prestige = 500
	}
	
	effect = {
		#Done as event to prevent RC formation event from firing
		show_as_tooltip = { form_frankish_kingdom_effect = yes }
		hidden_effect = {
			set_global_variable = {
				name = formed_frankish_kingdom
				value = yes
			}
			
			root = { save_scope_as = the_frankish_former }
			
			root = { trigger_event = { id = BP_claim_kingdom_decisions.0010 } }
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 100
		modifier = { #AI Hazkert will not do this
			add = -10000
			root = character:159590
		}
	}
}
