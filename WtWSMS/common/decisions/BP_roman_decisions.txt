### Major Decisions
ask_for_hre = {
	picture = {
		reference = "gfx/interface/illustrations/activity_header_backgrounds/activity_coronation.dds"
	}
	decision_group_type = major
	
	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 0
		empire = 60
		hegemony = 0
	}
	
	desc = ask_for_hre_desc
	selection_tooltip = ask_for_hre_tooltip
	
	is_shown = {
		#Haven't already asked
		root = { NOT = { has_variable = asked_for_hre } }
		#West Rome must not exist
		NOT = { exists = title:h_western_roman_empire.holder }
		NOT = { exists = title:e_wre.holder }
		NOT = { exists = title:e_hre.holder }
		#Have to be in communion with the Pope to do this
		faith = {
			OR = {
				this = faith:nicene
				this.religious_head = faith:nicene.religious_head
			}
		}
		age >= 16
		primary_title.tier = tier_empire
		#Only Non-Romans would dare broach this subject this way
		culture = { has_ability_to_reform_rome_directly_trigger = no }
		#Can't be of a specially excluded title
		NOR = {
			#Can't already be Roman Empire
			has_title = title:e_wre
			has_title = title:e_byzantium
			has_title = title:h_western_roman_empire
			has_title = title:h_eastern_roman_empire
			has_title = title:h_roman_empire
			#Axum is special
			has_title = title:e_axum
		}
	}
	
	is_valid = {
		gold >= 500
		custom_tooltip = {
			text = primary_title_must_be_empire
			primary_title.tier = tier_empire
		}
		realm_size >= 255
		custom_description = {
			text = must_control_at_least_33_percent_of_wre_territory
			subject = root
			title:h_western_roman_empire = {
				any_county = {
					percent >= 0.33
					this.holder.top_liege = root
				}
			}
		}
		OR = {
			has_trait = legendary_leader
			AND = {
				prestige_level >= 5
				legitimacy_level >= 5
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_adult = yes
		is_at_war = no
	}
	
	cost = {
		gold = massive_gold_value
	}
	
	effect = {
		root = {
			set_variable = { name = asked_for_hre }
			save_scope_as = empire_petitoner
		}
		this.faith.religious_head = { trigger_event = { id = BP_roman_decisions.0001 days = { 5 17 } } }
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 10 #%10 chance
		#Zealous and Ambitious required, also needs enough gold
		
		#Won't do without enough gold
		modifier = {
			factor = 0
			NOT = { gold >= massive_gold_value }
		}
		modifier = {
			factor = 2.5
			has_trait = cynical
		}
		modifier = {
			factor = 2.5
			has_trait = ambitious
		}
	}
}
ask_for_wre = {
	picture = {

	}
	decision_group_type = major
	
	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0

		empire = 60
		hegemony = 0
	}
	
	desc = ask_for_wre_desc
	selection_tooltip = ask_for_wre_tooltip
	
	is_shown = {

		root = { NOT = { has_variable = asked_for_wre } }
		#WRE must not exist
		NOT = { exists = title:h_western_roman_empire.holder }

		OR = {
			exists = title:e_byzantium.holder

		}

		OR = {
			religion = religion:christianity_religion

			#Can't already be Roman or other

		}
		realm_size >= 300 #Regardless of the size of Rome, it needs to be *big*;
		#Control 80% of whatever remains of Western Rome
		custom_description = {
			text = must_control_at_least_80_percent_of_wre_territory

			title:h_western_roman_empire = {
				any_county = {

			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_adult = yes
		is_at_war = no
	}
	
	cost = {
		gold = 1500
	}
	
	effect = {
		root = {
			set_variable = { name = asked_for_wre }
			save_scope_as = empire_petitoner
		}
		if = {
			limit = { exists = title:h_eastern_roman_empire.holder }
			title:h_eastern_roman_empire.holder = { trigger_event = { id = BP_roman_decisions.0004 days = { 5 17 } } } #The emperor hears the request
		}
		else = {
			title:e_byzantium.holder = { trigger_event = { id = BP_roman_decisions.0004 days = { 5 17 } } } #The emperor hears the request
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 10 #%10 chance
		#Zealous and Ambitious required, also needs enough gold

			NOT = { gold >= 500 }

		}

			factor = 2.5
			has_trait = ambitious
		}
	}

sole_roman_empire = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/roman_emperor_salute.dds"
	}
	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 0
		empire = 60
		hegemony = 0
	}
	
	desc = sole_roman_empire_desc
	selection_tooltip = sole_roman_empire_tooltip
	
	is_shown = {
		#Can't claim to be the sole empire if the other half (or its rump) exists
		trigger_if = {
			limit = {
				OR = {
					has_title = title:e_wre
					has_title = title:h_western_roman_empire
				}
			}
			NOT = { exists = title:e_byzantium.holder }
			NOT = { exists = title:h_eastern_roman_empire.holder }
		}
		trigger_else_if = {
			limit = {
				OR = {
					has_title = title:e_byzantium
					has_title = title:h_eastern_roman_empire
				}
			}
			NOT = { exists = title:e_wre.holder }
			NOT = { exists = title:h_western_roman_empire.holder }
		}
		trigger_else = { always = no }
		OR = {
			religion = religion:christianity_religion
			religion = religion:hellenism_religion
		}
		NOT = { has_claim_on = title:h_roman_empire }
		culture = { has_ability_to_reform_rome_directly_trigger = yes }
		NOT = { has_global_variable = sole_roman_empire }
	}
	
	is_valid = {
		gold >= 1000
		OR = {
			has_trait = legendary_leader
			AND = {
				prestige_level >= 5
				legitimacy_level >= 5
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_adult = yes
		is_at_war = no
	}
	
	cost = {
		gold = 1000
		prestige = 1500
	}
	
	effect = {
		set_global_variable = { name = sole_roman_empire }
		show_as_tooltip = { add_pressed_claim = title:h_roman_empire }
		custom_tooltip = { text = roman_empire_claim_allows_for_conquest }
		root = { trigger_event = { id = BP_roman_decisions.0010 } }
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 100 #%100 chance
		
		#Won't do without enough gold
		modifier = {
			factor = 0
			NOT = { gold >= 1000 }
		}
	}
}

### Minor Decisions
support_greens = {
	picture = {
		reference = "gfx/interface/illustrations/tournament_contest_illustrations/chariots_win_green.dds"
	}

	ai_check_interval = 60
	desc = support_greens_desc
	selection_tooltip = support_greens_tooltip
	
	is_shown = {
		root.top_liege = { any_realm_province = { has_building_with_flag = chariot_races } }
		AND = {
			primary_title.tier >= tier_county
			is_ruler = yes
		}
		NOR = {
			has_trait = greens_supporter
			has_trait = blues_supporter
		}
		
	}
	
	is_valid = {
		piety >= 50
	}
	
	is_valid_showing_failures_only = {
		is_available_adult = yes
	}
	
	cost = { piety = 50 }
	
	effect = {
		add_trait = greens_supporter
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 10 #%10 chance
		
		modifier = { #Not for the Zealous
			factor = 0
			NOT = { has_trait = zealous }
		}
		modifier = { #Cynical are more likely to
			factor = 1.2
			has_trait = cynical
		}
		modifier = { #We like the get buddy-buddy with our liege
			factor = 1.2
			liege = { has_trait = greens_supporter }
		}
	}
}
