subroman_chance_vassal_gets_aristocratic_law_improve_one_level = {
	value = 30
	
	#Bonuses
	
	##Influence & Prestige outstripping the liege
	if = {
		limit = { this.influence_level = this.top_liege.influence_level }
		add = 5
	}
	else_if = {
		limit = { this.influence_level > this.top_liege.influence_level }
		add = 10
	}
	if = {
		limit = { this.prestige_level = this.top_liege.prestige_level }
		add = 5
	}
	else_if = {
		limit = { this.prestige_level > this.top_liege.prestige_level }
		add = 10
	}
	
	#March of time
	if = {
		limit = { this.culture = { has_innovation = innovation_mundium } }
		add = 5
	}
	if = {
		limit = { this.culture = { has_innovation = innovation_hereditary_rule } }
		add = 10
	}
	if = {
		limit = { this.culture = { has_innovation = innovation_aristocracy } }
		add = 10
	}
	if = {
		limit = { this.culture = { has_innovation = innovation_baliffs } }
		add = 15
	}
	if = {
		limit = { this.culture = { has_innovation = innovation_royal_prerogative } }
		add = 15
	}
	
	#And just being in the eras themselves makes it more likely
	if = {
		limit = {
			this.culture = {
				num_discovered_innovations_in_era = { target = culture_era_late_antiquity value > 0 }
			}
		}
		add = 5
	}
	if = {
		limit = {
			this.culture = {
				num_discovered_innovations_in_era = { target = culture_era_early_medieval value > 0 }
			}
		}
		add = 10
	}
	
	#Maluses
	
	#Intimidated/cowed vassals are kept in line
	if = {
		limit = { this = { has_dread_level_towards = { target = this.top_liege level > 0 } } }
		add = -5
	}
	if = {
		limit = { this = { has_dread_level_towards = { target = this.top_liege level > 1 } } }
		add = -10
	}
	
	#Fighting the March of Time:  Roman Administration
	if = {
		limit = { this.culture = { has_innovation = innovation_roman_administration } }
		add = -10
	}
	if = {
		limit = { this.culture = { has_innovation = innovation_legal_codex } }
		add = -15
	}
}

subroman_vassal_with_law_cutoff_percent = 0.5 #Must be one greater than this number
number_of_subroman_administrative_dukes = {
	value = 0
	every_vassal = {
		limit = {
			primary_title.tier >= tier_duchy
			bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
		}
		add = 1
	}
}
percent_of_subroman_administrative_dukes_with_subroman_admin_aristocracy_law_level_0 = {
	value = 0
	every_vassal = {
		limit = {
			primary_title.tier >= tier_duchy
			bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
			has_realm_law_flag = subroman_admin_aristocracy_law_0
		}
		add = 1
	}
	divide = number_of_subroman_administrative_dukes
}
