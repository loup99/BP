clerical_bureau_director_influence_base = { #1.5% per skill level; it's a multiplier, not an addition
	add = stewardship
	multiply = 1.5
}

clerical_bureau_director_influence_max = {
	add = clerical_bureau_director_influence_base
	if = {
		limit = { councillor_liege_has_erudition_legacy_5_perk = yes }
		add = clerical_bureau_director_influence_erudition_bonus
	}
	if = {
		limit = { councillor_liege_has_family_business_bonus = yes }
		add = clerical_bureau_director_influence_family_business_bonus
	}
	if = {
		limit = { councillor_liege_has_consulted_house_bonus = yes }
		add = clerical_bureau_director_influence_consulted_house_bonus
	}
	if = {
		limit = {
			scope:councillor_liege = {
				has_title = title:c_byzantion
				title:c_byzantion = { has_county_modifier = pandidakterion_university_modifier }
			}
		}
		add = clerical_bureau_director_influence_pandidakterion_bonus
	}
}

clerical_bureau_director_influence_monthly_increase = {
	add = chancellor_domestic_affairs_opinion_max
	divide = 24
}

clerical_bureau_director_influence_scale = {
	value = 0
	if = {
		limit = {
			scope:councillor_liege = {
				is_ai = yes
				top_liege = this
			}
		}
		add = clerical_bureau_director_influence_max
	}
	else_if = {
		limit = {
			scope:councillor_liege = { has_variable = clerical_bureau_director_influence_value }
		}
		add = scope:councillor_liege.var:clerical_bureau_director_influence_value
	}
}
