#### Unification War Stuff
unification_war_fame_base = major_prestige_value

#Counts up number of available subjugations the player gets
total_available_subjugations = {
	value = 1
	if = { #One more if your religion is the right type
		limit = { this.faith = { has_doctrine_parameter = increased_subjugation_limit } }
		add = 1
	}
	if = { #One less if your religion is the wrong type
		limit = { this.faith = { has_doctrine_parameter = decreased_subjugation_limit } }
		subtract = 1
	}
	if = { #One more if your culture is the right type
		limit = { this.culture = { has_cultural_parameter = increased_subjugation_limit } }
		add = 1
	}
	if = { #One less if your culture is the wrong type
		limit = { this.culture = { has_cultural_parameter = decreased_subjugation_limit } }
		subtract = 1
	}
	if = { #One for Legendary Characters, given the right government
		limit = {
			OR = {
				this = { government_has_flag = government_is_tribal }
				this = { government_has_flag = government_is_clan }
			}
			
			any_character_trait = { has_trait_flag = increased_subjugation_limit }
		}
		add = 1
	}
	min = 1
}
