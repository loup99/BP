minority_capacity_of_county = {
		# Every county can support minorities as a function of its development; We will utilize the Lucas
		# Sequence Vn(1,−1) [the Lucas Numbers, a series related to the Fibonacci sequence]
		# 1, 3, 4, 7, 11, 18, 29, 47, 76 ...
	value = 1
	if = {
		limit = { scope:the_county.title_province = { NOT = { has_holding_type = tribal_holding } } }
		if = { limit = { scope:the_county.development_level >= 04 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 07 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 11 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 18 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 29 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 47 } add = 1 }
		if = { limit = { scope:the_county.development_level >= 76 } add = 1 }
	}
}
