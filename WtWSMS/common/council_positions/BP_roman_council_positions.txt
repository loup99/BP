#Major Roman Offices
councillor_chancellor_roman = {
	name = councillor_chancellor_roman
	tooltip = councillor_chancellor_roman_tooltip
	
	valid_position = {
		bp_has_access_to_roman_ministry_trigger = yes
	}

	can_fire = {
		NOT = { has_realm_law = roman_chancellor_law }
	}

	can_reassign = {
		NOT = { has_realm_law = roman_chancellor_law }
	}

	
	auto_fill = {
	}

	modifier = {
		name = councillor_chancellor_roman_modifier
		fellow_vassal_opinion = 5
		dread_baseline_add = 2
		monthly_diplomacy_lifestyle_xp_gain_mult = 0.05
		monthly_stewardship_lifestyle_xp_gain_mult = 0.02
		monthly_intrigue_lifestyle_xp_gain_mult = 0.02
		monthly_learning_lifestyle_xp_gain_mult = 0.02
		scale = council_scaled_by_liege_tier
	}

	modifier = { #Money-based scaling
		name = councillor_chancellor_roman_modifier
		monthly_prestige = 1
		monthly_influence = 1
		scale = council_scaled_monthly_income
	}

	modifier = { #Admin-based scaling
		name = councillor_chancellor_roman_modifier
		monthly_influence = 2
		enemy_political_scheme_success_chance_max_add = -5
		owned_scheme_secrecy_add = 5
		scale = council_scaled_admin_value
	}

	council_owner_modifier = {
		name = learn_on_the_job_modifier
		diplomacy = 1
		scale = chancellor_learn_on_the_job_scale
	}

	council_owner_modifier = {
		name = learn_on_the_job_modifier
		stewardship = 1
		scale = chancellor_learn_on_the_job_scale
	}

	council_owner_modifier = {
		name = reliable_house_modifier
		diplomacy = 1
		scale = chancellor_reliable_house_scale
	}

	council_owner_modifier = {
		name = reliable_house_modifier
		stewardship = 1
		scale = chancellor_reliable_house_scale
	}

	valid_character = {
		exists = root.liege_or_court_owner
		bp_can_be_roman_minister_trigger = { COURT_OWNER = root.liege_or_court_owner }
		tgp_is_ceremonial_liege_trigger = no
	}

	on_get_position = {
		got_council_position_effect = yes
		if = {
			limit = {
				scope:councillor_liege.primary_title.tier = tier_hegemony
				OR = {
					scope:councillor_liege = { government_has_flag = government_is_roman_administrative }
					scope:councillor_liege = { government_has_flag = government_is_subroman_administrative }
				}
			}
			every_empire = {
				limit = {
					this.hegemony = scope:councillor_liege.primary_title
					this = { has_variable = is_supreme_roman_ministry_title }
				}
				save_scope_as = the_ministry_title
				root = {
					got_roman_administrative_minister_position_effect = {
						MINISTER_TITLE = scope:the_ministry_title
						MINISTER_POSITION = councillor_chancellor_roman
					}
				}
			}
		}
	}

	on_fired_from_position = {
		fired_from_council_position_effect = yes
	}

	on_lose_position = {
		fired_roman_minister_position_effect = yes
	}

	portrait_animation = chancellor

	barbershop_data = {
		position = { 918.0 172.0 }
		click_to_front = no
	}
}
