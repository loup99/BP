#NB: only looks back 5 generations
any_ancestor_of_dynasty_trigger = {
	custom_description = {
		text = any_ancestor_of_dynasty_trigger_text
		subject = $DYNASTY$
		any_ancestor = { dynasty = $DYNASTY$ }
	}
}
#NB: Must be wrapped in a county scope; otherwise evaluates **all** provinces
has_a_non_tribal_holding_trigger = { #County Scope
	any_county_province = {
		OR = {
			has_holding_type = castle_holding
			has_holding_type = city_holding
			has_holding_type = church_holding
		}
	}
}
has_a_rough_terrain_province_trigger = { #County Scope
	any_county_province = {
		OR = {
			terrain = hills
			terrain = mountains
			terrain = desert_mountains
		}
	}
}
has_all_rough_terrain_provinces_trigger = { #County Scope
	any_county_province = {
		percent = 1.00
		OR = {
			terrain = hills
			terrain = mountains
			terrain = desert_mountains
		}
	}
}
