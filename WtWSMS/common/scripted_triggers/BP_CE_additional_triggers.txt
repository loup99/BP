additional_parent_child_checks_trigger = {
	additional_parent_culture_trigger = {
		CHARACTER = $CHARACTER$
		CULTURE = $CULTURE$
		CULTURE_IN_QUESTION = $CULTURE_IN_QUESTION$
	}
	additional_child_culture_trigger = {
		CHARACTER = $CHARACTER$
		CULTURE = $CULTURE$
		CULTURE_IN_QUESTION = $CULTURE_IN_QUESTION$
	}
}

additional_child_culture_trigger = {
	trigger_if = { #Additional Parent Cultures
		limit = { $CHARACTER$.culture = { has_variable_list = bastard_cultures } }
		custom_description = {
			text = culture_is_not_child_culture
			NOR = {
				$CHARACTER$.culture = { is_target_in_variable_list = { name = bastard_cultures target = flag:$CULTURE_IN_QUESTION$ } }
				$CULTURE$ = culture:$CULTURE_IN_QUESTION$
			}
		}
	}
}

additional_parent_culture_trigger = {
	trigger_if = {
		limit = { $CULTURE$ = { has_variable_list = bastard_cultures } }
		custom_description = {
			text = culture_is_not_parent_culture
			NOR = {
				$CULTURE$ = { is_target_in_variable_list = { name = bastard_cultures target = flag:$CULTURE_IN_QUESTION$ } }
				$CHARACTER$.culture = culture:$CULTURE_IN_QUESTION$
			}
		}
	}
}
