##### Changed from Vanilla: Added faith-based access to runestones
fp1_can_raise_stele_trigger = {
	OR = {
		culture = { has_cultural_parameter = has_access_to_runestones }
		faith = { has_doctrine_parameter = has_access_to_runestones }
		has_character_modifier = embraced_local_traditions_modifier
	}
	culture = {
		NOT = { has_cultural_era_or_later = culture_era_high_medieval }
	}
}

##### Changed from Vanilla: Compares native language instead of checking for North Germanic to determine stele effectiveness
fp1_can_raise_stele_title_scope_trigger = {
	culture = {
		has_same_culture_language = root.culture
		NOT = { has_cultural_era_or_later = culture_era_high_medieval }
	}
}

##### Changed from Vanilla: Added early heritage
fp1_eligible_for_yearly_events_trigger = {
	has_fp1_dlc_trigger = yes
	OR = {
		culture = {
			OR = {
				has_cultural_pillar = heritage_north_germanic
				has_cultural_pillar = heritage_scandzan
			}
		}
		religion = religion:germanic_religion
	}
}
fp1_is_norse = {
	OR = {
		culture = {
			OR = {
				has_cultural_pillar = heritage_north_germanic
				has_cultural_pillar = heritage_scandzan
			}
		}
		culture = culture:norman
	}
}

fp1_achievement_culture_norse_trigger = {
	fp1_is_norse = yes
}
