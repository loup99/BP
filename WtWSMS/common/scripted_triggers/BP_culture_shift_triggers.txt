is_a_living_culture_trigger = {
	this = { save_temporary_scope_as = the_culture }
	OR = {
		any_living_character = { this.culture = scope:the_culture }
		any_county = { this.culture = scope:the_culture }
		any_county = { county_has_specific_minority_culture_trigger = { CULTURE = scope:the_culture } }
	}
}

#Linguistic Drift Triggers
valid_linguistic_drift_culture_trigger = {
	this = {
		is_a_living_culture_trigger = yes
		#Types of languages supported: Latin
		has_cultural_pillar = language_latin
	}
}

valid_heritage_drift_culture_trigger = {
	this = {
		is_a_living_culture_trigger = yes
		#Types of heritages supported: Roman Heritage
		has_cultural_pillar = heritage_latin
	}
}
