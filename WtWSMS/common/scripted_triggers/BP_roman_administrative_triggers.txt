bp_has_access_to_roman_ministry_trigger = {
	government_has_flag = is_a_roman_administrative_government
	highest_held_title_tier >= tier_hegemony
}
bp_can_be_roman_minister_trigger = {
	can_be_councillor_basics_trigger = yes
	# Has to be roman administrative or a non-ruler
	OR = {
		is_ruler = no
		AND = {
			is_landless_ruler = yes
			government_has_flag = is_a_roman_administrative_government
		}
	}
	# Gender is based on laws of the liege
	OR = {
		AND = {
			is_female = yes
			$COURT_OWNER$ = {
				NOR = {
					has_realm_law = male_only_law
					has_realm_law = male_preference_law
				}
			}
		}
		AND = {
			is_male = yes
			$COURT_OWNER$ = {
				NOR = {
					has_realm_law = female_only_law
					has_realm_law = female_preference_law
				}
			}
		}
	}
}
