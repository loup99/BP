BP_sacred_eunuch_clothing_trigger = {
	OR = {
		AND = { # Sacred Eunuchs doctrine
			is_eunuch_trigger = yes
			faith = { has_doctrine = special_doctrine_sacred_eunuchs }
			OR = {
				portrait_religious_devoted_trigger = yes
				portrait_religious_high_trigger = yes
				is_clergy = yes
			}
		}
		AND = { # Scythian enarei
			faith = { has_doctrine_parameter = utilizes_enarei }
			has_trait = enarei
		}
	}
}
