#Teachings of the Sramanas
is_nastika_faith = {
	group = "special"
	
	is_available_on_create = {
		has_doctrine = special_doctrine_is_nastika_faith
	}

	special_doctrine_is_nastika_faith = {
		# Aniconism effects
		character_modifier = {
			church_holding_build_speed = -0.33
			church_holding_build_gold_cost = -0.33
			church_holding_holding_build_speed = -0.33
			church_holding_holding_build_gold_cost = -0.33
		}
		parameters = {
			hostility_same_religion = 0
			hostility_override_special_doctrine_humanistic_philosophy = 1
			take_vows_active = yes
			reforging_artifacts_costs_piety = yes
			destroying_artifacts_is_pious = yes
		}
	}
}
