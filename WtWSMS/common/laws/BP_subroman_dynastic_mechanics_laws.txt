#Somewhat unique:  you can't have better than your liege, but it's not locked to the liege's
subroman_dynastic_aristocracy_law = {
	default = subroman_dynastic_aristocracy_law_0
	cumulative = no
	
	flag = subroman_admin_aristocracy_law
	
	subroman_dynastic_aristocracy_law_0 = {
		modifier = {
			#monthly_influence_mult = 0.0
			title_creation_cost_mult = 0.0
			monthly_prestige_gain_mult = 0.0
		}
		flag = succession_weight_dynastic_1
		flag = governor_efficency_weight_1
		flag = subroman_admin_aristocracy_law_0
		
		can_keep = {
			bp_realm_law_use_subroman_dynastic_aristocracy_trigger = yes
		}
		
		can_pass = {
			trigger_if = {
				limit = { top_liege != this }
				custom_tooltip = {
					text = mechanically_driven_law
					always = no
				}
			}
			trigger_else = {
				custom_description = {
					subject = root
					text = "has_subroman_dynastic_aristocracy_law_cooldown"
					NAND = {
						has_variable = subroman_dynastic_aristocracy_law_cooldown
						NOT = { culture = { has_innovation = innovation_all_things } }
					}
				}
				#Adjacency
				has_realm_law = subroman_dynastic_aristocracy_law_1
				#At least 50 percent plus one of your vassals are at this law or better
				custom_tooltip = {
					text = half_of_subroman_dynastic_duke_vassals_has_law_0_or_better
					root.percent_of_subroman_administrative_dukes_with_subroman_admin_aristocracy_law_level_0 > root.subroman_vassal_with_law_cutoff_percent
				}
				#Trying to improve the law means you'll need some prestige and influence
				trigger_if = {
					limit = {
						has_realm_law = subroman_dynastic_aristocracy_law_1
						is_ai = no
					}
					prestige_level >= 3
					#influence_level >= 3
				}
			}
		}
		
		on_pass = {
			if = {
				limit = { top_liege != this }
				
			}
			else = {
				set_variable = {
					name = subroman_dynastic_aristocracy_law_cooldown
					years = @subroman_dynastic_aristocracy_law_cooldown
				}
