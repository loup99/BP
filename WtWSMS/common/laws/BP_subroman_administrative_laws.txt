subroman_admin_aristocracy_law = {
	default = subroman_admin_aristocracy_law_0
	cumulative = no
	
	flag = subroman_admin_aristocracy_law
	
	subroman_admin_aristocracy_law_0 = {
		modifier = {
			monthly_influence_mult = 0.0
			title_creation_cost_mult = 0.0
			monthly_prestige_gain_mult = 0.0
		}
		flag = succession_weight_dynastic_1
		flag = governor_efficency_weight_1
		
		can_keep = {
			bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
		}
		
		can_pass = {
			trigger_if = {
				limit = { top_liege != this }
				custom_tooltip = {
					text = mechanically_driven_law
					always = no
				}
			}
			trigger_else = {
				custom_description = {
					subject = root
					text = "has_subroman_admin_aristocracy_law_cooldown"
					NAND = {
						has_variable = subroman_admin_aristocracy_law_cooldown
						NOT = { culture = { has_innovation = innovation_all_things } }
					}
				}
				#Adjacency
				has_realm_law = subroman_admin_aristocracy_law_1
				#At least 50 percent plus one of your vassals are at this law or better
				custom_tooltip = {
					text = half_of_subroman_duke_vassals_has_law_0_or_better
					root.percent_of_subroman_administrative_dukes_with_subroman_admin_aristocracy_law_level_0 > root.subroman_vassal_with_law_cutoff_percent
				}
				#Trying to improve the law means you'll need some prestige and influence
				trigger_if = {
					limit = {
						has_realm_law = subroman_admin_aristocracy_law_1
						is_ai = no
					}
					prestige_level >= 3
					influence_level >= 3
				}
			}
		}
		
		on_pass = {
			if = {
				limit = { top_liege != this }
				
			}
			else = {
				set_variable = {
					name = subroman_admin_aristocracy_law_cooldown
					years = @subroman_admin_aristocracy_law_cooldown
				}
				if = {
					limit = { has_realm_law = subroman_admin_aristocracy_law_1 }
					every_vassal = { #Decline Law One Level
						limit = {
							bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
							NOT = { has_realm_law = subroman_admin_aristocracy_law_0 }
							NOT = { has_realm_law = subroman_admin_aristocracy_law_4 }
						}
						random = {
							chance = subroman_chance_vassal_gets_aristocratic_law_decline_one_level
							decline_aristocracy_law_one_level_effect = yes
							send_interface_message = {
								type = event_generic_neutral
								title = top_liege_makes_aristocracy_less_powerful_tt
								desc = top_liege_makes_aristocracy_less_powerful_desc
							}
						}
					}
				}
			}
		}
		
		ai_will_do = {
			#AI will improve to this law if they're running short on prestige or influence
			if = {
				limit = { has_realm_law = subroman_admin_aristocracy_law_1 }
				if = {
					limit = { this.prestige < 0 }
					value = 100
				}
				else_if = {
					limit = { this.influence < 50 }
					value = 50
				}
				else = { value = 0 }
			}
		}
	}

			monthly_influence_mult = 0.10

					}
				}

					value = 100
				}
				else_if = {
					limit = { this.influence < 50 }
					value = 50
				}
				else = { value = 0 }
			}

			}

					root.percent_of_subroman_administrative_dukes_with_subroman_admin_aristocracy_law_level_2 > root.subroman_vassal_with_law_cutoff_percent
				}
				#Trying to improve the law means you'll need some prestige and influence
				trigger_if = {

						has_realm_law = subroman_admin_aristocracy_law_3
						is_ai = no
					}

					influence_level >= 3
				}
			}
		}
		
		on_pass = {
			if = {
				limit = { top_liege != this }
				
			}
			else = {
				set_variable = {
					name = subroman_admin_aristocracy_law_cooldown
					years = @subroman_admin_aristocracy_law_cooldown
				}
				if = {
					limit = { has_realm_law = subroman_admin_aristocracy_law_1 }
					add_prestige = 150
					change_influence = 150
					
					every_vassal = { #Improve Law One Level
						limit = {
							bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
							NOT = { has_realm_law = subroman_admin_aristocracy_law_4 }
						}
						random = {
							chance = subroman_chance_vassal_gets_aristocratic_law_improve_one_level
							improve_aristocracy_law_one_level_effect = yes
							send_interface_message = {
								type = event_generic_neutral
								title = top_liege_makes_aristocracy_more_powerful_tt
								desc = top_liege_makes_aristocracy_more_powerful_desc
							}
						}
					}
				}
				else_if = {
					limit = { has_realm_law = subroman_admin_aristocracy_law_3 }
					every_vassal = { #Decline Law One Level
						limit = {
							bp_realm_law_use_subroman_administrative_aristocracy_trigger = yes
							NOT = { has_realm_law = subroman_admin_aristocracy_law_0 }
							NOT = { has_realm_law = subroman_admin_aristocracy_law_4 }
						}
						random = {
							chance = subroman_chance_vassal_gets_aristocratic_law_decline_one_level
							decline_aristocracy_law_one_level_effect = yes
							send_interface_message = {
								type = event_generic_neutral
								title = top_liege_makes_aristocracy_less_powerful_tt
								desc = top_liege_makes_aristocracy_less_powerful_desc
							}
						}
					}
				}
			}
		}
		
		#AI tries to improve to this law when it's influence is too low
		ai_will_do = {
			#AI will improve to this law if they're running short on prestige or influence
			if = {
				limit = { has_realm_law = subroman_admin_aristocracy_law_3 }
				if = {
					limit = { this.prestige < 0 }
					value = 100
				}
				else_if = {
					limit = { this.influence < 50 }
					value = 50
				}
				else = { value = 0 }
			}
			#AI will decline to this the law when their children are not in succession
			if = {
				limit = { has_realm_law = subroman_admin_aristocracy_law_1 }
				if = {
					limit = { NOT = { any_child = { is_primary_heir_of = root } } }
					value = 100
				}
				else = { value = 0 }
			}
		}
	}
