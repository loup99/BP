@bureaucracy_cooldown_years = 25

bureaucratic_organization = {
	default = bureaucratic_organization_civil
	cumulative = no
	flag = other_law_category
	flag = bureaucratic_law
	
	bureaucratic_organization_civil = {
		modifier = {}
		
		can_keep = { realm_law_use_bureaucratic_other_laws_trigger = yes }
		
		can_pass = {
			is_independent_ruler = yes
			trigger_if = {
				custom_description = {
					subject = root
					text = "has_bureaucratic_organization_cooldown"
					NOT = { has_variable = bureaucracy_organization_cooldown }
				}
			}
		}
		
		on_pass = {
			set_variable = {
				name = bureaucracy_organization_cooldown
				years = @bureaucracy_cooldown_years
			}
		}
	}
