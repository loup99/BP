@bureaucracy_cooldown_years = 25

#Organization models the overall orientation of the state in the shifts of Roman/Byzantine governance between
#split civil/military organization and combined civil/military organization.
bureaucratic_organization = {
	default = bureaucratic_organization_civil
	cumulative = no
	flag = other_law_category
	flag = bureaucratic_organization
	
	bureaucratic_organization_civil = {
		modifier = {}
		
		can_keep = { has_government = bureaucratic_government }
		can_have = { has_government = bureaucratic_government }
		
		can_pass = {
			is_independent_ruler = yes
			trigger_if = {
				custom_description = {
					subject = root
					text = "has_bureaucratic_organization_cooldown"
					NOT = { has_variable = bureaucracy_organization_cooldown }
				}
			}
		}
		
		on_pass = {
			set_variable = {
				name = bureaucracy_organization_cooldown
				years = @bureaucracy_cooldown_years
			}
			custom_description = {
				text = all_gubernatorials_get_law
				apply_law_down_tree_effect = { LAW = gubernatorial_organization_civil }
			}
		}
	}
	bureaucratic_organization_mixed = {
		modifier = {}
		
		can_keep = { has_government = bureaucratic_government }
