roman_diarchy_law = {
	default = free_diarchy_law
	cumulative = no
	flag = imperial_policy

	free_diarchy_law = {
		flag = free_diarchy_law
		
		can_have = {
			bp_has_access_to_roman_ministry_trigger = yes
		}

		can_pass = {
			custom_tooltip = {
				text = admin_law_cooldown_desc
				NOT = { has_variable = admin_law_cooldown }
			}
			#Other Diarchy Requirements
			trigger_if = {
				limit = { has_realm_law = roman_chancellor_law }
				custom_tooltip = {
					text = current_diarch_must_be_weak_or_loyal
					any_vassal = {
						any_held_title = { has_variable = is_supreme_roman_ministry_title }
						save_temporary_scope_as = the_diarch
					}
					NOT = { scope:the_diarch ?= { has_hook = root } }
					OR = {
						root.diarch ?= { diarch_loyalty >= 75 }
						root = { diarchy_swing < 5 }
					}
				}
			}
			trigger_if = {
				limit = { has_realm_law = roman_marshal_law }
				custom_tooltip = {
					text = current_diarch_must_be_weak_or_loyal
					any_vassal = {
						any_held_title = { has_variable = is_supreme_roman_military_title }
						save_temporary_scope_as = the_diarch
					}
					NOT = { scope:the_diarch ?= { has_hook = root } }
					OR = {
						root.diarch ?= { diarch_loyalty >= 75 }
						root = { diarchy_swing < 5 }
					}
				}
			}
		}

		on_pass = {
			## Set cooldown
			diarch = { depose_diarch = yes }
			set_variable = {
				name = admin_law_cooldown
				years = @admin_law_cooldown_value
			}
		}

		#Roman Bureaucracy does not like a free type of diarchy
		modifier = {
			roman_administrative_government_vassal_opinion = -10
			roman_administrative_government_tax_contribution_mult = -0.25
			roman_administrative_government_levy_contribution_mult = -0.25
			enemy_hostile_scheme_success_chance_max_add = 5
			monthly_influence_mult = -0.1
			close_relative_opinion = 10
		}

		ai_will_do = {
			value = 1
		}
	}
