				if = {
					limit = {
						has_doctrine = eastern_hostility_doctrine
						NOT = { has_doctrine = tenet_ethnic_religion }
						NOT = { this = scope:the_county_temp.faith }
					}
					add_to_variable_list = {
						name = faith_minorities_small
						target = scope:the_faith_temp
					}
				}
			}
		}
	}
