five_year_playable_pulse = {
	on_actions = {
		delay = { days = { 30 1095 } } #one month to 3 years
		BP_feudal_mutation
		BP_admin_aristocracy_solidification
	}
}

BP_feudal_mutation = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		government_has_flag = government_is_roman_federate
		#Maximal transitions
		NOT = { vassal_contract_has_flag = succession_is_purely_dynastic }
	}
	random_events = {
		chance_to_happen = 7 #7% every 5 years implies a bit less than 67% chance every 75 years
		100 = BP_admin_to_feudal_events.0050
	}
}

#Solidification of the Aristocracy as a hereditary institution
BP_admin_aristocracy_solidification = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		government_has_flag = government_is_roman_federate
		NOT = { vassal_contract_has_flag = succession_is_purely_dynastic }
		NOT = { has_realm_law = confederate_partition_succession_law }
	}
	random_events = {
		chance_to_happen = 7 #7% every 5 years implies a bit less than 67% chance every 75 years
		100 = BP_admin_to_feudal_events.0100
	}
}
