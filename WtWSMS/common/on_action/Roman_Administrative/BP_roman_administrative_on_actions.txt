on_title_gain = {
	events = {}
	on_actions = {
		BP_got_a_roman_ministry
		BP_keep_roman_federate_government
	}
}

BP_got_a_roman_ministry = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = {
			limit = {
				scope:title.tier >= tier_empire
				scope:title = { has_variable = is_a_roman_ministry_title }
			}
			scope:title.de_jure_liege.holder = { save_scope_as = councillor_liege }
			scope:title = { save_scope_as = the_ministry_title }
			switch = {
				trigger = has_variable
				is_supreme_roman_military_title = {
					got_roman_administrative_minister_position_effect = {
						MINISTER_TITLE = scope:the_ministry_title
						MINISTER_POSITION = councillor_marshal_roman
					}
				}
				is_supreme_roman_ministry_title = {
					got_roman_administrative_minister_position_effect = {
						MINISTER_TITLE = scope:the_ministry_title
						MINISTER_POSITION = councillor_chancellor_roman
					}
				}
			}
		}
		#Set Up Roman Premiership if possible
		this.liege_or_court_owner ?= { establish_premiership_if_necessary_effect = yes }
	}
}
BP_keep_roman_federate_government = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		is_independent_ruler = no
	}
	effect = {
		if = {
			limit = {
				
				scope:title = { has_variable = roman_federate_title }
				scope:previous_holder = { has_government = roman_federate_government }
			}
			change_government = roman_federate_government
		}
	}
}

on_vassal_change = {
	events = {}
	on_actions = {
		BP_vassal_changed_roman_admin
	}
}

BP_vassal_changed_roman_admin = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = {
			limit = {
				# independent title
				OR = {
					liege ?= { bp_has_access_to_roman_ministry_trigger = no }
					this = top_liege
				}
				any_held_title = {
					this.tier = tier_empire
					this = { has_variable = is_supreme_roman_military_title }
				}
			}
			every_held_title = {
				if = {
					limit = { this = { has_variable = is_supreme_roman_military_title } }
					root = { destroy_title = prev }
				}
				if = {
					limit = { this = { has_variable = is_supreme_roman_ministry_title } }
					root = { destroy_title = prev }
				}
			}
		}
	}
}

on_title_lost = {
	events = {}
	on_actions = {
		BP_flip_federate_to_admin
	}
}
BP_flip_federate_to_admin = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = {
			limit = { is_landless_ruler = yes }
			if = {
				limit = { root.top_liege = { has_government = roman_administrative_government } }
				change_government = roman_administrative_government
			}
		}
	}
}
