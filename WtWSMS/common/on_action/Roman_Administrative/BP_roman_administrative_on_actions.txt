on_title_gain = {
	events = {}
	on_actions = {
		BP_got_a_roman_ministry
		BP_keep_roman_federate_government
		BP_flip_federate_when_not_roman
	}
}

BP_got_a_roman_ministry = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = {
			limit = {
				scope:title.tier >= tier_empire
				scope:title = { has_variable = is_a_roman_ministry_title }
			}
			scope:title.de_jure_liege.holder = { save_scope_as = councillor_liege }
			scope:title = { save_scope_as = the_ministry_title }
			switch = {
				trigger = has_variable
				is_supreme_roman_military_title = {
					got_roman_administrative_minister_position_effect = {
						MINISTER_TITLE = scope:the_ministry_title
						MINISTER_POSITION = councillor_marshal_roman
					}
				}
				is_supreme_roman_ministry_title = {
					got_roman_administrative_minister_position_effect = {
						MINISTER_TITLE = scope:the_ministry_title
						MINISTER_POSITION = councillor_chancellor_roman
					}
				}
			}
		}
		#Set Up Roman Premiership if possible
		this.liege_or_court_owner ?= { establish_premiership_if_necessary_effect = yes }
	}
}
BP_keep_roman_federate_government = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		is_independent_ruler = no
	}
	effect = {
		if = {
			limit = { scope:previous_holder ?= { has_government = roman_federate_government } }
			change_government = roman_federate_government
		}
	}
}
BP_flip_federate_when_not_roman = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		has_government = roman_federate_government
	}
	effect = {
		if = {
			limit = { NOT = { top_liege = { government_has_flag = is_a_roman_government } } }
			if = {
				limit = { root.culture = { has_cultural_era_or_later = culture_era_classical_antiquity } }
				change_government = high_tribal_government
			}
			else = { change_government = tribal_government }
		}
	}
}

on_vassal_change = {
	events = {}
	on_actions = {
		BP_independent_ministry_cleanup
		BP_independent_roman_government_checks
	}
}
BP_independent_ministry_cleanup = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = { # Destroy independent ministry empire titles
			limit = {
				OR = {
					liege ?= { bp_has_access_to_roman_ministry_trigger = no }
					this = top_liege
				}
				any_held_title = {
					this.tier = tier_empire
					OR = {
						this = { has_variable = is_supreme_roman_military_title }
						this = { has_variable = is_supreme_roman_ministry_title }
					}
				}
			}
			#Destroy Ministries
			every_held_title = {
				if = {
					limit = { this = { has_variable = is_supreme_roman_military_title } }
					root = { destroy_title = prev }
				}
				if = {
					limit = { this = { has_variable = is_supreme_roman_ministry_title } }
					root = { destroy_title = prev }
				}
			}
		}
	}
}
BP_independent_roman_government_checks = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = { #Independent Roman Admin governments must be kingdoms, or they become subroman
			limit = {
				government_has_flag = government_only_valid_for_kingdoms_empires
				root = top_liege
				primary_title.tier < tier_kingdom
			}
			change_government = roman_administrative_government
		}
		#Independent Federates go tribal/high tribal
		if = {
			limit = {
				government_has_flag =  government_becomes_tribal_high_tribal_if_independent_or_not_under_roman
				root = top_liege
			}
			if = {
				limit = { root.culture = { has_cultural_era_or_later = culture_era_classical_antiquity } }
				change_government = high_tribal_government
			}
			else = { change_government = tribal_government }
		}
	}
}

on_title_lost = {
	events = {}
	on_actions = {
		BP_flip_federate_to_admin_when_landless
	}
}
BP_flip_federate_to_admin_when_landless = {
	trigger = { BP_has_dlc_necessary_for_roman_governments_trigger = yes }
	effect = {
		if = {
			limit = { is_landless_ruler = yes }
			if = {
				limit = { root.top_liege = { has_government = roman_administrative_government } }
				change_government = roman_administrative_government
			}
			else_if = {
				limit = { root.top_liege = { has_government = subroman_administrative_government } }
				change_government = subroman_administrative_government
			}
		}
	}
}
