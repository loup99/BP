five_year_playable_pulse = {
	on_actions = {
		delay = { days = { 30 1095 } } #one month to 3 years
		BP_admin_to_feudal_mutation
		BP_admin_to_feudal_aristocracy_solidification
		BP_admin_to_feudal_military_landing
		BP_admin_to_feudal_domestic_taxation
	}
}

BP_admin_to_feudal_mutation = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		bp_meets_requirements_to_fedualize_trigger = yes
	}
	random_events = {
		chance_to_happen = 7 #7% every 5 years implies a bit less than 67% chance every 75 years
		100 = BP_admin_to_feudal_events.0050
	}
}

BP_admin_to_feudal_aristocracy_solidification = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes

on_war_won_attacker = {
	on_actions = {
		delay = { days = { 7 28 } } #one to four weeks
		BP_admin_to_fedual_military_landing_post_war
	}
}

#When we _win_ wars we have a chance of having landing fire off
#As is per-war instead of per 5 year block, it's a 50% chance every 10 wars won
BP_admin_to_fedual_military_landing_post_war = {
	trigger = {
		BP_has_dlc_necessary_for_roman_governments_trigger = yes
		scope:attacker.top_liege = this
	}
	effect = {
		random_vassal = {
			limit = { bp_general_feudalization_event_trigger = yes }
			trigger_event = { on_action = BP_admin_to_feudal_military_landing }
		}
	}
}
