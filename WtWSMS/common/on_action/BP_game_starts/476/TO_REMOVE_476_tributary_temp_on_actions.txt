on_game_start = { on_actions = { bp_game_476_temp_tributaries } }

bp_game_476_temp_tributaries = {
	trigger = { current_date = 476.2.2 }
	on_actions = {
		BP_apply_tributary_status_e_byzantium
		BP_apply_tributary_status_e_persia
		BP_apply_tributary_status_e_xiongnu
		BP_apply_tributary_status_e_rouran
		BP_apply_tributary_status_k_vandalica
		BP_apply_tributary_status_k_soissons
	}
}
BP_apply_tributary_status_e_byzantium = {
	effect = {
		#Build the list of tributaries
		title:c_zichia = { add_to_temporary_list = byz_tributaries }
		title:d_thracian_goths = { add_to_temporary_list = byz_tributaries }
		title:k_salihid = { add_to_temporary_list = byz_tributaries }
		title:k_arabia_petraea = { add_to_temporary_list = byz_tributaries }
		title:k_sarir = { add_to_temporary_list = byz_tributaries }
		title:k_lazica = { add_to_temporary_list = byz_tributaries }
		
		make_the_tributaries_effect = {
			LIEGE = title:e_byzantium.holder
			THE_LIST_OF_TRIBS = byz_tributaries
		}
	}
}

	on_actions = {
		delay = { days = 1 }
		BP_inherit_contract_modification_blocked
	}
}

BP_inherit_contract_modification_blocked = { #Inherited tributaries are blocked from modification...
	effect = {
		if = {
			limit = { vassal_contract_has_flag = tributary_applied }
			set_vassal_contract_modification_blocked = yes
		}
	}
}
