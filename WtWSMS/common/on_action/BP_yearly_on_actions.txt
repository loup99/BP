yearly_global_pulse = {
	on_actions = {
		liguistic_shift
		heritage_shift
	}
}

liguistic_shift = {
	effect = {
		generic_shift_culture_event = {
			SHIFT_TRIGGER = valid_linguistic_drift_culture_trigger
			EVENT_ID = bp_heritage_linguistic_drift.0001
			HERITAGE_OR_LANGUAGE = language
		}
	}
}

heritage_shift = {
	effect = {
		generic_shift_culture_event = {
			SHIFT_TRIGGER = valid_heritage_drift_culture_trigger
			EVENT_ID = bp_heritage_linguistic_drift.0101
			HERITAGE_OR_LANGUAGE = heritage
		}
	}
}
