#Global Triggers
scripted_trigger basic_game_requirements_for_transition_trigger = {
	#CK2 was game start +50, but this won't work well with later bookmarks
	#So using game_start+10 (to prevent from firing _too_ early, with events
	#applicable after AD 500
	current_date >= 450.1.1
	years_from_game_start > 10
}
scripted_trigger basic_character_requirements_for_transition_trigger = {
	OR = {
		government_has_flag = government_is_roman_federate
		government_has_flag = government_is_subroman_administrative
		government_has_flag = government_is_subroman_dynastic
	}
	primary_title = { tier > tier_barony }
}

#Scripted Effects
scripted_effect protect_religious_rights_if_needed = {
	#Continue Religious rights if necessary
	save_temporary_scope_as = the_char
	if = {
		limit = {
			scope:the_char.top_liege.faith = {
				faith_hostility_level = { target = scope:the_char.faith value > 0 }
			}
		}
		vassal_contract_set_obligation_level = {
			type = religious_rights
			level = religious_rights_protected
		}
	}
}
scripted_effect shift_government_effect = {
	hidden_effect = {
		remove_realm_law_if_has_it_effect = { LAW = roman_federate_appointment_succession_law }
		remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_0 }
		remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_1 }
		remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_2 }
		remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_3 }
		remove_realm_law_if_has_it_effect = { LAW = roman_dynastic_authority_0 }
		remove_realm_law_if_has_it_effect = { LAW = roman_dynastic_authority_1 }
		remove_realm_law_if_has_it_effect = { LAW = roman_dynastic_authority_2 }
		remove_realm_law_if_has_it_effect = { LAW = roman_dynastic_authority_3 }
	}
	
	switch = {
		trigger = has_character_flag
		
		transition_to_feudal_government = {
			change_government = feudal_government
			hidden_effect = {
				add_law_if_not_has_it_effect = { LAW = confederate_partition_succession_law }
				add_law_if_not_has_it_effect = { LAW = crown_authority_0 }
			}
			remove_character_flag = transition_to_feudal_government
			#Continue Religious rights if necessary
			protect_religious_rights_if_needed = yes
		}
		transition_to_clan_government = {
			change_government = clan_government
			hidden_effect = {
				add_law_if_not_has_it_effect = { LAW = clan_impassive_partition_succession_law }
				add_law_if_not_has_it_effect = { LAW = crown_authority_0 }
			}
			remove_character_flag = transition_to_clan_government
			#Continue Religious rights if necessary
			protect_religious_rights_if_needed = yes
		}
		transition_to_high_tribal_government = {
			change_government = high_tribal_government
			hidden_effect = {
				add_law_if_not_has_it_effect = { LAW = clan_impassive_partition_succession_law }
				add_law_if_not_has_it_effect = { LAW = crown_authority_0 }
			}
			remove_character_flag = transition_to_high_tribal_government
			#Continue Religious rights if necessary
			protect_religious_rights_if_needed = yes
		}
		transition_to_tribal_government = {
			change_government = tribal_government
			hidden_effect = {
				add_law_if_not_has_it_effect = { LAW = confederate_partition_succession_law }
				add_law_if_not_has_it_effect = { LAW = tribal_authority_0 }
			}
			remove_character_flag = transition_to_tribal_government
		}
	}
}
scripted_effect more_feudalization_effect = {
	if = {
		limit = { bp_meets_requirements_to_fedualize_trigger = yes }
		trigger_event = { id = BP_admin_to_feudal_events.0060 days = { 3 7 } }
	}
	else_if = {
		limit = {
			OR = {
				government_has_flag = government_is_roman_federate
				government_has_flag = government_is_subroman_administrative
				government_has_flag = government_is_subroman_dynastic
				NOT = { has_character_flag = had_a_feudal_transition_event }
			}
		}
		if = {
			limit = { is_ai = no }
			send_interface_message = {
				type = msg_governor_duties
				title = feudal_transition_accelerates_tt
				desc = feudal_transition_accelerates_desc
			}
		}
		add_character_flag = {
			flag = had_a_feudal_transition_event
			years = had_a_feudal_transition_event_cooldown_years
		}
		#All the rights shift instead, and attach variables
		increase_hereditary_rights_one_level_effect = yes
		this.primary_title = { set_variable = aristocracy_solidification_cooldown }
		this = { set_variable = aristocracy_solidification_cooldown }
		increase_military_landing_one_level_effect = yes
		this.primary_title = { set_variable = military_landing_cooldown }
		this = { set_variable = military_landing_cooldown }
		decay_domestic_taxation_one_level_effect = yes
		this.primary_title = { set_variable = taxation_decay_cooldown }
		this = { set_variable = taxation_decay_cooldown }
	}
}
scripted_effect vassals_transiton_government_effect = {
	#every sub-roman converts to Feudal
	every_vassal_or_below = {
		limit = { government_has_flag = government_is_subroman_administrative_or_dynastic }
		if = {
			limit = { this.top_liege = { government_has_flag = government_is_feudal } }
			add_character_flag = transition_to_feudal_government
			shift_government_effect = yes
		}
		else_if = {
			limit = { this.top_liege = { government_has_flag = government_is_clan } }
			add_character_flag = transition_to_clan_government
			shift_government_effect = yes
		}
		else = {
			add_character_flag = transition_to_feudal_government
			shift_government_effect = yes
		}
	}
	every_vassal_or_below = {
		limit = { government_has_flag = government_is_roman_federate }
		if = { #If they meet the limit to feudalize, they get to be clan
			limit = { bp_meets_requirements_to_fedualize_trigger = yes }
			if = {
				limit = { character_can_adopt_feudal_government_trigger = yes }
				add_character_flag = transition_to_feudal_government
				shift_government_effect = yes
			}
			else_if = {
				limit = { character_can_adopt_clan_government_trigger = yes }
				add_character_flag = transition_to_clan_government
				shift_government_effect = yes
			}
			else = {
				add_character_flag = transition_to_feudal_government
				shift_government_effect = yes
			}
		}
		else_if = {
			limit = { valid_for_federate_to_be_high_tribal_trigger = yes }
			add_character_flag = transition_to_high_tribal_government
			shift_government_effect = yes
		}
		else = {
			add_character_flag = transition_to_tribal_government
			shift_government_effect = yes
		}
	}
	
}
scripted_effect neighbors_shift_more_to_feudalism_effect = {
	every_neighboring_top_liege_realm = {
		this.holder = { more_feudalization_effect = yes }
	}
}
#Scripted Triggers
scripted_trigger character_can_adopt_clan_government_trigger = {
	OR = {
		this.faith = { has_doctrine_parameter = right_to_rule_clan }
		this.culture = { has_cultural_parameter = right_to_rule_clan }
	}
}
scripted_trigger character_can_adopt_feudal_government_trigger = {
	OR = {
		this.faith = { has_doctrine_parameter = right_to_rule_feudal }
		this.culture = { has_cultural_parameter = right_to_rule_feudal }
	}
}

#The actual feudalization chain
BP_admin_to_feudal_events.0050 = { #Hidden Event; sets up everything
	type = character_event
	hidden = yes
	
	trigger = {
		basic_game_requirements_for_transition_trigger = yes
		bp_meets_requirements_to_fedualize_trigger = yes
	}
	
	immediate = {
		save_scope_as = the_key_feudalizing_char
		add_character_flag = { flag = the_key_feudalizing_char days = 30 }
		trigger_event = { id = BP_admin_to_feudal_events.0060 days = { 3 7 } }
	}
}
#Federate transition to Feudalism
BP_admin_to_feudal_events.0060 = {
	type = character_event
	title = BP_admin_to_feudal_events.0060.title
	desc = BP_admin_to_feudal_events.0060.desc
	
	theme = realm
	window = fullscreen_event
	override_background = BP_feudal_mutation
	content_source = dlc_WtWSMS
	
	right_portrait = {
		character = ROOT
		animation = steward
	}
	
	immediate = {
		save_scope_as = the_feudal_mutation_char
		if = {
			limit = { character_can_adopt_clan_government_trigger = yes }
			add_character_flag = transition_to_clan_government
		}
		if = {
			limit = { character_can_adopt_feudal_government_trigger = yes }
			add_character_flag = transition_to_feudal_government
		}
		else = {
			add_character_flag = transition_to_feudal_government
		}
	}
	
	option = {
		trigger = { character_can_adopt_clan_government_trigger = yes }
		name = BP_admin_to_feudal_events.0060.clan
		
		
		shift_government_effect = yes
		vassals_transiton_government_effect = yes
		if = {
			limit = { is_independent_ruler = yes }
			neighbors_shift_more_to_feudalism_effect = yes
		}
		
		#Inform the liege if the key triggering character is the char
		if = {
			limit = { root = { has_character_flag = the_key_feudalizing_char } }
			every_liege_or_above = { trigger_event = { id = BP_admin_to_feudal_events.0061 days = { 3 7 } } }
		}
	}
	option = {
		trigger = { character_can_adopt_feudal_government_trigger = yes }
		fallback = yes
		name = BP_admin_to_feudal_events.0060.feudal
		
		shift_government_effect = yes
		vassals_transiton_government_effect = yes
		if = {
			limit = { is_independent_ruler = yes }
			neighbors_shift_more_to_feudalism_effect = yes
		}
		#Inform the liege if the key triggering character is the char
		if = {
			limit = { root = { has_character_flag = the_key_feudalizing_char } }
			every_liege_or_above = { trigger_event = { id = BP_admin_to_feudal_events.0061 days = { 3 7 } } }
		}
	}
}

			limit = {

				NOT = { has_relation_rival = $KEY_CHAR$.top_liege }
			}
			random_list = {
				1 = {
					set_relation_rival = $KEY_CHAR$.top_liege
					if = {
						limit = { has_relation_grudge = $KEY_CHAR$.top_liege }
						remove_relation_grudge = $KEY_CHAR$.top_liege
					}
					modifier = {
						factor = 0
						NOT = { has_relation_potential_rival = $KEY_CHAR$.liege }
					}
				}
				1 = {
					set_relation_potential_rival = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						has_relation_potential_rival = $KEY_CHAR$.liege
					}
				}
				1 = {
					set_relation_grudge = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						OR = {
							has_relation_potential_rival = $KEY_CHAR$.top_liege
							has_relation_rival = $KEY_CHAR$.top_liege
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				NOT = { has_trait = content }
				NOT = { has_relation_rival = $KEY_CHAR$.top_liege }
			}
			random_list = {
				1 = {
					set_relation_rival = $KEY_CHAR$.top_liege
					if = {
						limit = { has_relation_grudge = $KEY_CHAR$.top_liege }
						remove_relation_grudge = $KEY_CHAR$.top_liege
					}
					modifier = {
						factor = 0
						NOT = { has_relation_potential_rival = $KEY_CHAR$.liege }
					}
				}
				1 = {
					set_relation_potential_rival = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						has_relation_potential_rival = $KEY_CHAR$.liege
					}
				}
				1 = {
					set_relation_grudge = $KEY_CHAR$.top_liege

						factor = 0
						OR = {
							has_relation_potential_rival = $KEY_CHAR$.top_liege
