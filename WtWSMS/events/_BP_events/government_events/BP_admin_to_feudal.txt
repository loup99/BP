#Global Triggers
scripted_trigger basic_game_requirements_for_transition_trigger = {
	#CK2 was game start +50, but this won't work well with later bookmarks
	#So using game_start+10 (to prevent from firing _too_ early, with events
	#applicable after AD 500
	current_date >= 450.1.1
	#years_from_game_start > 10
}
scripted_trigger basic_character_requirements_for_transition_trigger = {
	government_has_flag = government_is_roman_federate
	primary_title = { tier > tier_barony }
}
#Global Effects
scripted_effect increase_hereditary_rights_one_level_effect = {
	if = {
		limit = { vassal_contract_has_flag = succession_weight_dynastic_1 }
		vassal_contract_set_obligation_level = { type = roman_federate_aristocracy level = roman_federate_aristocracy_1 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = succession_weight_dynastic_2 }
		vassal_contract_set_obligation_level = { type = roman_federate_aristocracy level = roman_federate_aristocracy_2 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = succession_weight_dynastic_3 }
		vassal_contract_set_obligation_level = { type = roman_federate_aristocracy level = roman_federate_aristocracy_3 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = succession_weight_dynastic_4 }
		vassal_contract_set_obligation_level = { type = roman_federate_aristocracy level = roman_federate_aristocracy_4 }
	}
}
scripted_effect advance_hereditary_rights_effect = {
	if = {
		limit = { NOT = { $CHAR$ = { vassal_contract_has_flag = succession_is_purely_dynastic } } }
		$CHAR$ = { increase_hereditary_rights_one_level_effect = yes }
	}
	#If you're already at max, you get confed partition
	#NB: this is *not* required for feudalism, but can be stacked to it
	else = {
		$CHAR$ = { add_realm_law_skip_effects = confederate_partition_succession_law }
	}
}
scripted_effect increase_military_landing_one_level_effect = {
	if = {
		limit = { vassal_contract_has_flag = roman_federate_military_landed_0 }
		vassal_contract_set_obligation_level = { type = roman_federate_military_landed level = roman_federate_military_landed_1 }

		vassal_contract_set_obligation_level = { type = roman_federate_military_landed level = roman_federate_military_landed_2 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = roman_federate_military_landed_2 }
		vassal_contract_set_obligation_level = { type = roman_federate_military_landed level = roman_federate_military_landed_3 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = roman_federate_military_landed_3 }
		vassal_contract_set_obligation_level = { type = roman_federate_military_landed level = roman_federate_military_landed_4 }
	}
	else_if = {
		limit = { vassal_contract_has_flag = roman_federate_military_landed_4 }
		vassal_contract_set_obligation_level = { type = roman_federate_military_landed level = roman_federate_military_landed_5 }
	}

scripted_effect set_feudal_transition_cooldown_effect = {
	add_character_flag = {
		flag = had_a_feudal_transition_event
		years = had_a_feudal_transition_event_cooldown_years
	}
	add_character_flag = {
		flag = $SPECIFIC_TYPE$
		years = feudalization_component_specific_cooldown_years
	}
	this.primary_title ?= {
		set_variable = {
			name = $SPECIFIC_TYPE$
			years = feudalization_component_specific_cooldown_years
		}
	}
}

#Scripted Effects
scripted_effect shift_government_effect = {
	remove_realm_law_if_has_it_effect = { LAW = roman_federate_appointment_succession_law }
	remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_0 }
	remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_1 }
	remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_2 }
	remove_realm_law_if_has_it_effect = { LAW = imperial_bureaucracy_3 }
	
	switch = {
		trigger = has_character_flag
		
		transition_to_feudal_government = {
			change_government = feudal_government
			add_law_if_not_has_it_effect = { LAW = confederate_partition_succession_law }
			add_law_if_not_has_it_effect = { LAW = crown_authority_0 }
			remove_character_flag = transition_to_feudal_government
		}
		transition_to_clan_government = {
			change_government = clan_government
			add_law_if_not_has_it_effect = { LAW = clan_impassive_partition_succession_law }
			add_law_if_not_has_it_effect = { LAW = crown_authority_0 }
			remove_character_flag = transition_to_clan_government
		}
	}
}
scripted_effect more_feudalization_effect = {
	if = {
		limit = { bp_meets_requirements_to_fedualize_trigger = yes }
		trigger_event = { id = BP_admin_to_feudal_events.0060 }
	}
	else_if = {
		limit = { government_has_flag = government_is_roman_federate }
		add_character_flag = {
			flag = had_a_feudal_transition_event
			years = had_a_feudal_transition_event_cooldown_years
		}
		#All the rights shift instead, and attach variables
		increase_hereditary_rights_one_level_effect = yes
		this.primary_title = { set_variable = aristocracy_solidification_cooldown }
		this = { set_variable = aristocracy_solidification_cooldown }
		increase_military_landing_one_level_effect = yes
		this.primary_title = { set_variable = military_landing_cooldown }
		this = { set_variable = military_landing_cooldown }
	}
}
scripted_effect further_feudalization_of_vassals_neighbors_effect = {
	every_vassal = { more_feudalization_effect = yes }
	#Spreads out to neighbors
	every_realm_county = {
		every_neighboring_county = {
			limit = { NOT = { this.holder = $KEY_CHAR$ } }
			this.holder = { more_feudalization_effect = yes }
		}
	}
}

BP_admin_to_feudal_events.0060 = {
	type = character_event
	title = BP_admin_to_feudal_events.0060.title
	desc = BP_admin_to_feudal_events.0060.desc
	
	theme = stewardship_domain_focus
	content_source = dlc_WtWSMS
	
	right_portrait = {
		character = ROOT
		animation = steward
	}
	
	immediate = {
		save_scope_as = the_feudal_mutation_char
	}
	
	option = {
		trigger = {
			OR = {
				root.faith = { has_doctrine_parameter = right_to_rule_clan }
				root.culture = { has_cultural_parameter = right_to_rule_clan }
			}
		}
		name = BP_admin_to_feudal_events.0060.clan
		
		add_character_flag = transition_to_clan_government
		shift_government_effect = yes
		further_feudalization_of_vassals_neighbors_effect = { KEY_CHAR = scope:the_feudal_mutation_char }
		
		#Inform the liege if the key triggering character is the char
		if = {
			limit = { root = { has_character_flag = the_key_feudalizing_char } }
			every_liege_or_above = { trigger_event = { id = BP_admin_to_feudal_events.0061 } }
		}
	}
	option = {
		trigger = {
			OR = {
				root.faith = { has_doctrine_parameter = right_to_rule_feudal }
				root.culture = { has_cultural_parameter = right_to_rule_feudal }
			}
		}
		fallback = yes
		name = BP_admin_to_feudal_events.0060.feudal
		
		add_character_flag = transition_to_feudal_government
		shift_government_effect = yes
		further_feudalization_of_vassals_neighbors_effect = { KEY_CHAR = scope:the_feudal_mutation_char }
		
		#Inform the liege if the key triggering character is the char
		if = {
			limit = { root = { has_character_flag = the_key_feudalizing_char } }
			every_liege_or_above = { trigger_event = { id = BP_admin_to_feudal_events.0061 } }
		}
	}
}

#Scripted Effects
scripted_effect apply_opinions_for_action = {
	$KEY_CHAR$ = {
		add_opinion = {
			modifier = subroman_$TYPE$_my_hereditary_customs
			target = $KEY_CHAR$.top_liege
		}
	}
	$KEY_CHAR$.top_liege = {
		every_vassal_or_below = {
			limit = {
				government_has_flag = government_is_roman_federate
				NOT = { this = $KEY_CHAR$ }
			}
			add_opinion = {
				modifier = subroman_$TYPE$_others_hereditary_customs
				target = $KEY_CHAR$.top_liege
			}
		}
	}
}
scripted_effect build_grudge_rivalry_effect = {
	$KEY_CHAR$ = {
		if = {
			limit = {
				has_trait = ambitious
				NOT = { has_relation_rival = $KEY_CHAR$.top_liege }
			}
			random_list = {
				1 = {
					set_relation_rival = $KEY_CHAR$.top_liege
					if = {
						limit = { has_relation_grudge = $KEY_CHAR$.top_liege }
						remove_relation_grudge = $KEY_CHAR$.top_liege
					}
					modifier = {
						factor = 0
						NOT = { has_relation_potential_rival = $KEY_CHAR$.liege }
					}
				}
				1 = {
					set_relation_potential_rival = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						has_relation_potential_rival = $KEY_CHAR$.liege
					}
				}
				1 = {
					set_relation_grudge = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						OR = {
							has_relation_potential_rival = $KEY_CHAR$.top_liege
							has_relation_rival = $KEY_CHAR$.top_liege
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				NOT = { has_trait = content }
				NOT = { has_relation_rival = $KEY_CHAR$.top_liege }
			}
			random_list = {
				1 = {
					set_relation_rival = $KEY_CHAR$.top_liege
					if = {
						limit = { has_relation_grudge = $KEY_CHAR$.top_liege }
						remove_relation_grudge = $KEY_CHAR$.top_liege
					}
					modifier = {
						factor = 0
						NOT = { has_relation_potential_rival = $KEY_CHAR$.liege }
					}
				}
				1 = {
					set_relation_potential_rival = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						has_relation_potential_rival = $KEY_CHAR$.liege
					}
				}
				1 = {
					set_relation_grudge = $KEY_CHAR$.top_liege
					modifier = {
						factor = 0
						OR = {
							has_relation_potential_rival = $KEY_CHAR$.top_liege
							has_relation_rival = $KEY_CHAR$.top_liege
						}
					}
				}
				1 = {}
			}
		}
	}
}
